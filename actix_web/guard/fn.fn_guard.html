use actix_web::{guard, web, App, HttpResponse};

fn main() {
    App::new().service(web::resource("/index.html").route(
        web::route()
            .guard(
                guard::fn_guard(
                    |req| req.headers()
                             .contains_key("content-type")))
            .to(|| HttpResponse::MethodNotAllowed()))
    );
}
