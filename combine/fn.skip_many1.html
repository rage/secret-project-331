Parses p one or more times ignoring the result.

NOTE: If p can succeed without consuming any input this may hang forever as skip_many1 will repeatedly use p to parse the same location in the input every time

let result = skip_many1(digit())
    .parse("123A");
assert_eq!(result, Ok(((), "A")));
