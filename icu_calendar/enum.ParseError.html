#[non_exhaustive]

    Range(RangeError),
    MissingFields,
    UnknownCalendar,
    MismatchedCalendar(CalendarAlgorithm, CalendarAlgorithm),
}
Expand description

An error returned from parsing an RFC 9557 string to an icu_calendar type.

This enum is marked as non-exhaustive
