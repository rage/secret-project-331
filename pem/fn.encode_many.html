pub fn encode_many(pems: &[Pem]) -> String
Expand description

Encode multiple PEM structs into a PEM-encoded data string

§Example

 use pem::{Pem, encode_many};

 let data = vec![
    Pem {
        tag: String::from("FOO"),
        contents: vec![1, 2, 3, 4],
    },
    Pem {
        tag: String::from("BAR"),
        contents: vec![5, 6, 7, 8],
    },
  ];
  encode_many(&data);
