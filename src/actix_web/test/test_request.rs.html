24/// Test `Request` builder.
25///
26/// For unit testing, actix provides a request builder type and a simple handler runner. TestRequest implements a builder-like pattern.
27/// You can generate various types of request via TestRequest's methods:
28/// - [`TestRequest::to_request`] creates an [`actix_http::Request`](Request).
29/// - [`TestRequest::to_srv_request`] creates a [`ServiceRequest`], which is used for testing middlewares and chain adapters.
30/// - [`TestRequest::to_srv_response`] creates a [`ServiceResponse`].
31/// - [`TestRequest::to_http_request`] creates an [`HttpRequest`], which is used for testing handlers.
32///
33/// ```
34/// use actix_web::{test, HttpRequest, HttpResponse, HttpMessage};
35/// use actix_web::http::{header, StatusCode};
36///
37/// async fn handler(req: HttpRequest) -> HttpResponse {
38///     if let Some(hdr) = req.headers().get(header::CONTENT_TYPE) {
39///         HttpResponse::Ok().into()
40///     } else {
41///         HttpResponse::BadRequest().into()
42///     }
43/// }
44///
45/// #[actix_web::test]
46/// # // force rustdoc to display the correct thing and also compile check the test
47/// # async fn _test() {}
48/// async fn test_index() {
49///     let req = test::TestRequest::default()
50///         .insert_header(header::ContentType::plaintext())
51///         .to_http_request();
52///
53///     let resp = handler(req).await;
54///     assert_eq!(resp.status(), StatusCode::OK);
55///
56///     let req = test::TestRequest::default().to_http_request();
57///     let resp = handler(req).await;
58///     assert_eq!(resp.status(), StatusCode::BAD_REQUEST);
59/// }
60/// ```
61pub struct TestRequest {
62    req: HttpTestRequest,
63    rmap: ResourceMap,
64    config: AppConfig,
65    path: Path<Url>,
66    peer_addr: Option<SocketAddr>,
67    app_data: Extensions,
68    #[cfg(feature = "cookies")]
69    cookies: CookieJar,
70}

74        TestRequest {
75            req: HttpTestRequest::default(),
76            rmap: ResourceMap::new(ResourceDef::new("")),

79            peer_addr: None,
80            app_data: Extensions::new(),
81            #[cfg(feature = "cookies")]

83        }
84    }
85}

232    /// Finalizes test request.
233    ///
234    /// This request builder will be useless after calling `finish()`.
235    fn finish(&mut self) -> Request {
236        // mut used when cookie feature is enabled
237        #[allow(unused_mut)]
238        let mut req = self.req.finish();
239
240        #[cfg(feature = "cookies")]
241        {
242            use actix_http::header::{HeaderValue, COOKIE};
243
244            let cookie: String = self
245                .cookies
246                .delta()
247                // ensure only name=value is written to cookie header
248                .map(|c| c.stripped().encoded().to_string())
249                .collect::<Vec<_>>()
250                .join("; ");
251
252            if !cookie.is_empty() {
253                req.headers_mut()
254                    .insert(COOKIE, HeaderValue::from_str(&cookie).unwrap());
255            }
256        }
257
258        req
259    }

261    /// Finalizes request creation and returns `Request` instance.

268    /// Finalizes request creation and returns `ServiceRequest` instance.
269    pub fn to_srv_request(mut self) -> ServiceRequest {
270        let (mut head, payload) = self.finish().into_parts();
271        head.peer_addr = self.peer_addr;
272        self.path.get_mut().update(&head.uri);
273
274        let app_state = AppInitServiceState::new(Rc::new(self.rmap), self.config.clone());
275
276        ServiceRequest::new(
277            HttpRequest::new(
278                self.path,
279                head,
280                app_state,
281                Rc::new(self.app_data),
282                None,
283                Default::default(),
284            ),
285            payload,
286        )
287    }
288
289    /// Finalizes request creation and returns `ServiceResponse` instance.
290    pub fn to_srv_response<B>(self, res: HttpResponse<B>) -> ServiceResponse<B> {
291        self.to_srv_request().into_response(res)

312    /// Finalizes request creation and returns `HttpRequest` and `Payload` pair.
313    pub fn to_http_parts(mut self) -> (HttpRequest, Payload) {
314        let (mut head, payload) = self.finish().into_parts();
315        head.peer_addr = self.peer_addr;
316        self.path.get_mut().update(&head.uri);
317
318        let app_state = AppInitServiceState::new(Rc::new(self.rmap), self.config.clone());
319
320        let req = HttpRequest::new(
321            self.path,
322            head,
323            app_state,
324            Rc::new(self.app_data),
325            None,
326            Default::default(),
327        );
328
329        (req, payload)
330    }

332    /// Finalizes request creation, calls service, and waits for response future completion.
333    pub async fn send_request<S, B, E>(self, app: &S) -> S::Response
334    where
335        S: Service<Request, Response = ServiceResponse<B>, Error = E>,
336        E: std::fmt::Debug,

355    #[actix_rt::test]
356    async fn test_basics() {
357        let req = TestRequest::default()
358            .version(Version::HTTP_2)
359            .insert_header(header::ContentType::json())
360            .insert_header(header::Date(SystemTime::now().into()))
361            .param("test", "123")
362            .data(10u32)
363            .app_data(20u64)
364            .peer_addr("127.0.0.1:8081".parse().unwrap())
365            .to_http_request();
366        assert!(req.headers().contains_key(header::CONTENT_TYPE));
367        assert!(req.headers().contains_key(header::DATE));
368        assert_eq!(
369            req.head().peer_addr,
370            Some("127.0.0.1:8081".parse().unwrap())
371        );
372        assert_eq!(&req.match_info()["test"], "123");
373        assert_eq!(req.version(), Version::HTTP_2);
374        let data = req.app_data::<Data<u32>>().unwrap();
375        assert!(req.app_data::<Data<u64>>().is_none());
376        assert_eq!(*data.get_ref(), 10);
377
378        assert!(req.app_data::<u32>().is_none());
379        let data = req.app_data::<u64>().unwrap();
380        assert_eq!(*data, 20);
381    }
