1#[cfg(test)]
2macro_rules! assert_simple_behaviour {
3    ($clear:expr, $secure:expr) => ({
4        assert_eq!($clear.iter().count(), 0);
5
6        $secure.add(Cookie::new("name", "val"));
7        assert_eq!($clear.iter().count(), 1);
8        assert_eq!($secure.get("name").unwrap().value(), "val");
9        assert_ne!($clear.get("name").unwrap().value(), "val");
10
11        $secure.add(Cookie::new("another", "two"));
12        assert_eq!($clear.iter().count(), 2);
13
14        $clear.remove(Cookie::named("another"));
15        assert_eq!($clear.iter().count(), 1);
16
17        $secure.remove(Cookie::named("name"));
18        assert_eq!($clear.iter().count(), 0);
19    })
20}

22#[cfg(test)]
23macro_rules! assert_secure_behaviour {
24    ($clear:expr, $secure:expr) => ({
25        $secure.add(Cookie::new("secure", "secure"));
26        assert!($clear.get("secure").unwrap().value() != "secure");
27        assert!($secure.get("secure").unwrap().value() == "secure");
28
29        let mut cookie = $clear.get("secure").unwrap().clone();
30        let new_val = format!("{}l", cookie.value());
31        cookie.set_value(new_val);
32        $clear.add(cookie);
33        assert!($secure.get("secure").is_none());
34
35        let mut cookie = $clear.get("secure").unwrap().clone();
36        cookie.set_value("foobar");
37        $clear.add(cookie);
38        assert!($secure.get("secure").is_none());
39    })
40}
