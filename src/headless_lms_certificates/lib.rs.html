28/**
29Generates a certificate as a png.
30
31## Arguments
32
33- debug: If true, the certificate will have a red dot at the anchoring points, and the URL will be replaced with a placeholder.
34  Should be false when rendering certificates for the students. However, when positioning the texts, this can be used to see why the texts were positioned where they were.
35*/
36#[allow(clippy::too_many_arguments)]
37pub async fn generate_certificate(
38    conn: &mut PgConnection,
39    file_store: &dyn FileStore,
40    certificate: &GeneratedCertificate,
41    debug: bool,
42    icu4x_blob: Icu4xBlob,
43) -> UtilResult<Vec<u8>> {
44    let config = headless_lms_models::certificate_configurations::get_by_id(
45        &mut *conn,
46        certificate.certificate_configuration_id,
47    )
48    .await
49    .map_err(|original_error| {
50        UtilError::new(
51            UtilErrorType::Other,
52            "Certificate configuration not found".to_string(),
53            Some(original_error.into()),
54        )
55    })?;
56    let background_svg = file_store
57        .fetch_file_content_or_use_filesystem_cache(Path::new(&config.background_svg_path))
58        .await?;
59    let overlay_svg = OptionFuture::from(
60        config
61            .overlay_svg_path
62            .as_ref()
63            .map(|path| file_store.fetch_file_content_or_use_filesystem_cache(Path::new(path))),
64    )
65    .await
66    .transpose()?;
67
68    let fontdb = font_loader::get_font_database_with_fonts(&mut *conn, file_store).await?;
69    let url = if debug {
70        "https://courses.mooc.fi/certificates/validate/debug".to_string()
71    } else {
72        format!(
73            // TODO: use base url here
74            "https://courses.mooc.fi/certificates/validate/{}",
75            certificate.verification_id
76        )
77    };
78    let date = if debug {
79        // TODO: this fixes the date for system tests, not a great solution...
80        NaiveDate::from_ymd_opt(2023, 1, 1).unwrap()
81    } else {
82        certificate.created_at.date_naive()
83    };
84    let texts_to_render = vec![
85        TextToRender {
86            text: certificate.name_on_certificate.to_string(),
87            y_pos: config.certificate_owner_name_y_pos,
88            x_pos: config.certificate_owner_name_x_pos,
89            font_size: config.certificate_owner_name_font_size,
90            text_anchor: config.certificate_owner_name_text_anchor,
91            text_color: config.certificate_owner_name_text_color,
92            ..Default::default()
93        },
94        TextToRender {
95            text: url,
96            y_pos: config.certificate_validate_url_y_pos,
97            x_pos: config.certificate_validate_url_x_pos,
98            font_size: config.certificate_validate_url_font_size,
99            text_anchor: config.certificate_validate_url_text_anchor,
100            text_color: config.certificate_validate_url_text_color,
101            ..Default::default()
102        },
103        TextToRender {
104            text: get_date_as_localized_string(&config.certificate_locale, date, icu4x_blob)?,
105            y_pos: config.certificate_date_y_pos,
106            x_pos: config.certificate_date_x_pos,
107            font_size: config.certificate_date_font_size,
108            text_anchor: config.certificate_date_text_anchor,
109            text_color: config.certificate_date_text_color,
110            ..Default::default()
111        },
112    ];
113    let paper_size = config.paper_size;
114
115    let res = generate_certificate_impl(
116        &background_svg,
117        overlay_svg.as_deref(),
118        &texts_to_render,
119        &paper_size,
120        debug,
121        Arc::new(fontdb),
122    )?;
123    Ok(res)
124}

126fn generate_certificate_impl(
127    background_svg: &[u8],
128    overlay_svg: Option<&[u8]>,
129    texts: &[TextToRender],
130    paper_size: &PaperSize,
131    debug_show_anchoring_points: bool,
132    fontdb: Arc<fontdb::Database>,
133) -> UtilResult<Vec<u8>> {
134    let start_setup = Instant::now();
135    let opt = usvg::Options {
136        font_family: "Lato".to_string(),
137        dpi: 600.0,
138        image_rendering: usvg::ImageRendering::OptimizeQuality,
139        shape_rendering: usvg::ShapeRendering::GeometricPrecision,
140        text_rendering: usvg::TextRendering::OptimizeLegibility,
141        fontdb,
142        ..Default::default()
143    };
144
145    let mut pixmap = resvg::tiny_skia::Pixmap::new(paper_size.width_px(), paper_size.height_px())
146        .ok_or_else(|| {
147        UtilError::new(
148            UtilErrorType::Other,
149            "Could not create a pixmap".to_string(),
150            None,
151        )
152    })?;
153    info!("Setup time {:?}", start_setup.elapsed());
154    let parse_background_svg_start = Instant::now();
155    let tree = usvg::Tree::from_data(background_svg, &opt).map_err(|original_error| {
156        UtilError::new(
157            UtilErrorType::Other,
158            "Could not parse background svg".to_string(),
159            Some(original_error.into()),
160        )
161    })?;
162
163    info!(
164        "Parse background svg time {:?}",
165        parse_background_svg_start.elapsed()
166    );
167
168    let start_render_background = Instant::now();
169    // Scaling the background to the paper size, if the aspect ratio is wrong (for example if it does not follow the aspect ratio of A4 paper size), the background will get stretched.
170    // If that's the case, the you should fix the background svg.
171    let background_size = tree.size().to_int_size();
172    let x_scale = paper_size.width_px() as f32 / background_size.width() as f32;
173    let y_scale = paper_size.height_px() as f32 / background_size.height() as f32;
174    info!(
175        "Background size {:?}, paper size: {:?}, x_scale: {}, y_scale: {}",
176        background_size, paper_size, x_scale, y_scale
177    );
178    resvg::render(
179        &tree,
180        resvg::tiny_skia::Transform::from_scale(x_scale, y_scale),
181        &mut pixmap.as_mut(),
182    );
183
184    info!(
185        "Render background time {:?}",
186        start_render_background.elapsed()
187    );
188
189    let text_svg_data = generate_text_svg(texts, debug_show_anchoring_points, paper_size)?;
190    info!("{}", String::from_utf8_lossy(&text_svg_data));
191    let parse_text_svg_start = Instant::now();
192    let text_tree = usvg::Tree::from_data(&text_svg_data, &opt).map_err(|original_error| {
193        UtilError::new(
194            UtilErrorType::Other,
195            "Could not parse text svg".to_string(),
196            Some(original_error.into()),
197        )
198    })?;
199
200    info!("Parse text svg time {:?}", parse_text_svg_start.elapsed());
201
202    let render_text_start = Instant::now();
203    resvg::render(
204        &text_tree,
205        tiny_skia::Transform::default(),
206        &mut pixmap.as_mut(),
207    );
208    info!("Render text time {:?}", render_text_start.elapsed());
209
210    if let Some(overlay_svg) = overlay_svg {
211        let start_render_overlay = Instant::now();
212        let overlay_tree = usvg::Tree::from_data(overlay_svg, &opt).map_err(|original_error| {
213            UtilError::new(
214                UtilErrorType::Other,
215                "Could not parse overlay svg".to_string(),
216                Some(original_error.into()),
217            )
218        })?;
219        resvg::render(
220            &overlay_tree,
221            tiny_skia::Transform::default(),
222            &mut pixmap.as_mut(),
223        );
224
225        info!("Overlay time {:?}", start_render_overlay.elapsed());
226    }
227
228    let save_png_start = Instant::now();
229    let png = pixmap.encode_png().map_err(|original_error| {
230        UtilError::new(
231            UtilErrorType::Other,
232            "Could convert pixmap to png".to_string(),
233            Some(original_error.into()),
234        )
235    })?;
236    info!("Save png time {:?}", save_png_start.elapsed());
237    Ok(png)
238}

240pub struct TextToRender {
241    pub text: String,
242    pub font_family: String,
243    pub font_size: String,
244    pub text_color: String,
245    pub x_pos: String,
246    pub y_pos: String,
247    /// How to align the text related to x_pos and y_pos. See: <https://developer.mozilla.org/en-US/docs/Web/SVG/Attribute/text-anchor>.
248    pub text_anchor: CertificateTextAnchor,
249}
250
251impl Default for TextToRender {
252    fn default() -> Self {
253        Self {
254            font_family: "Lato".to_string(),
255            font_size: "150px".to_string(),
256            text_color: "black".to_string(),
257            x_pos: "50%".to_string(),
258            y_pos: "50%".to_string(),
259            text: "Example text".to_string(),
260            text_anchor: CertificateTextAnchor::Middle,
261        }
262    }
263}

265fn generate_text_svg(
266    texts: &[TextToRender],
267    debug_show_anchoring_points: bool,
268    paper_size: &PaperSize,
269) -> UtilResult<Vec<u8>> {
270    let mut writer = Writer::new(Cursor::new(Vec::new()));
271    writer
272        .create_element("svg")
273        .with_attribute(("width", format!("{}px", paper_size.width_px()).as_str()))
274        .with_attribute(("height", format!("{}px", paper_size.height_px()).as_str()))
275        .with_attribute(("xmlns", "http://www.w3.org/2000/svg"))
276        .write_inner_content(|writer| {
277            for text in texts {
278                writer
279                    .create_element("text")
280                    .with_attribute(("x", text.x_pos.as_str()))
281                    .with_attribute(("y", text.y_pos.as_str()))
282                    .with_attribute((
283                        "style",
284                        format!(
285                            "font-size: {}; font-family: {}; fill: {}; font-weight: 400;",
286                            text.font_size, text.font_family, text.text_color
287                        )
288                        .as_str(),
289                    ))
290                    .with_attribute(("text-anchor", text.text_anchor.to_string().as_str()))
291                    .write_text_content(BytesText::from_escaped(&text.text))
292                    .map_err(|_original_error| {
293                        // Might not be optimal but that's the Error type of the closure that comes from the library and we don't want to unwrap here and potentially crash the process.
294                        io::Error::other("Could not write text to svg".to_string())
295                    })?;
296
297                if debug_show_anchoring_points {
298                    writer
299                        .create_element("circle")
300                        .with_attribute(("cx", text.x_pos.as_str()))
301                        .with_attribute(("cy", text.y_pos.as_str()))
302                        .with_attribute(("r", "5"))
303                        .with_attribute(("fill", "#da2e2e"))
304                        .write_empty()
305                        .map_err(|_original_error| {
306                            // Might not be optimal but that's the Error type of the closure that comes from the library and we don't want to unwrap here and potentially crash the process.
307                            io::Error::other("Could not write debug point to svg".to_string())
308                        })?;
309                    writer
310                        .create_element("circle")
311                        .with_attribute(("cx", text.x_pos.as_str()))
312                        .with_attribute(("cy", text.y_pos.as_str()))
313                        .with_attribute(("r", "3"))
314                        .with_attribute(("fill", "#ff9b9b"))
315                        .write_empty()
316                        .map_err(|_original_error| {
317                            // Might not be optimal but that's the Error type of the closure that comes from the library and we don't want to unwrap here and potentially crash the process.
318                            io::Error::other("Could not write debug point to svg".to_string())
319                        })?;
320                }
321            }
322            Ok(())
323        })
324        .map_err(|original_error: std::io::Error| {
325            UtilError::new(
326                UtilErrorType::Other,
327                "Could not write text svg".to_string(),
328                Some(original_error.into()),
329            )
330        })?;
331
332    Ok(writer.into_inner().into_inner())
333}
