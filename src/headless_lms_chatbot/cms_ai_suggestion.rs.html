19/// Structured LLM response for CMS paragraph suggestions.
20#[derive(serde::Deserialize)]
21struct CmsParagraphSuggestionResponse {
22    suggestions: Vec<String>,
23}
24
25/// System prompt for generating multiple alternative paragraph suggestions for CMS content.
26const SYSTEM_PROMPT: &str = r#"You are helping course staff improve a single paragraph of course material.
27
28Your task is to generate several alternative versions of the given paragraph based on the requested action.
29
30General rules:
31- Always preserve the original meaning and important details unless the action explicitly asks to add or remove content.
32- Maintain a clear, pedagogical tone appropriate for course materials.
33- Do not invent facts that contradict the original paragraph.
34
35About the suggestions:
36- Produce multiple alternative rewrites of the same paragraph.
37- Each suggestion must be meaningfully different in structure, emphasis, or level of detail.
38- Do NOT output suggestions that only differ by tiny edits (e.g. one or two word changes).
39- Keep each suggestion self-contained and suitable for direct insertion into the material.
40
41You will receive:
42- The original paragraph text.
43- The requested action (e.g. fix spelling, improve clarity, change tone, translate).
44- Optional metadata such as target tone and target language.
45
46Your output must follow the JSON schema exactly:
47{
48  "suggestions": ["...", "...", "..."]
49}"#;
50
51/// User prompt prefix; the concrete action and metadata will be appended.
52const USER_PROMPT_PREFIX: &str = "Generate multiple rewritten versions of the paragraph according to the requested action and metadata. The paragraph may contain inline HTML markup valid inside a Gutenberg paragraph; preserve existing inline tags (links, emphasis, code, sub/superscripts) where possible, do not introduce block-level elements, and do not add new formatting to spans of text that were previously unformatted. Return JSON only.";
53
54/// Input payload for CMS paragraph suggestions.
55pub struct CmsParagraphSuggestionInput {
56    pub action: String,
57    pub content: String,
58    pub is_html: bool,
59    pub meta_tone: Option<String>,
60    pub meta_language: Option<String>,
61    pub meta_setting_type: Option<String>,
62}
63
64/// Generate multiple paragraph suggestions for CMS using an LLM with structured JSON output.
65pub async fn generate_paragraph_suggestions(
66    app_config: &ApplicationConfiguration,
67    task_lm: TaskLMSpec,
68    input: &CmsParagraphSuggestionInput,
69) -> ChatbotResult<Vec<String>> {
70    let CmsParagraphSuggestionInput {
71        action,
72        content,
73        is_html: _,
74        meta_tone,
75        meta_language,
76        meta_setting_type,
77    } = input;
78
79    let mut system_instructions = SYSTEM_PROMPT.to_owned();
80    system_instructions.push_str("\n\nRequested action: ");
81    system_instructions.push_str(action);
82    if let Some(tone) = meta_tone {
83        system_instructions.push_str("\nTarget tone: ");
84        system_instructions.push_str(tone);
85    }
86    if let Some(lang) = meta_language {
87        system_instructions.push_str("\nTarget language: ");
88        system_instructions.push_str(lang);
89    }
90    if let Some(setting_type) = meta_setting_type {
91        system_instructions.push_str("\nSetting type: ");
92        system_instructions.push_str(setting_type);
93    }
94
95    let paragraph_source = content.as_str();
96
97    let user_message_content = format!(
98        "{prefix}\n\nOriginal paragraph (may include inline HTML):\n{paragraph}",
99        prefix = USER_PROMPT_PREFIX,
100        paragraph = paragraph_source
101    );
102
103    let used_tokens =
104        estimate_tokens(&system_instructions) + estimate_tokens(&user_message_content);
105    let token_budget =
106        calculate_safe_token_limit(task_lm.context_size, task_lm.context_utilization);
107
108    if used_tokens > token_budget {
109        return Err(ChatbotError::new(
110            ChatbotErrorType::ChatbotMessageSuggestError,
111            "Input paragraph is too long for the CMS AI suggestion context window.".to_string(),
112            None,
113        ));
114    }
115
116    let system_message = APIMessage {
117        role: MessageRole::System,
118        fields: APIMessageKind::Text(APIMessageText {
119            content: system_instructions,
120        }),
121    };
122
123    let user_message = APIMessage {
124        role: MessageRole::User,
125        fields: APIMessageKind::Text(APIMessageText {
126            content: user_message_content,
127        }),
128    };
129
130    let params = if task_lm.thinking {
131        LLMRequestParams::Thinking(ThinkingParams {
132            max_completion_tokens: Some(4000),
133            verbosity: None,
134            reasoning_effort: None,
135            tools: vec![],
136            tool_choice: None,
137        })
138    } else {
139        LLMRequestParams::NonThinking(NonThinkingParams {
140            max_tokens: Some(2000),
141            temperature: None,
142            top_p: None,
143            frequency_penalty: None,
144            presence_penalty: None,
145        })
146    };
147
148    let chat_request = LLMRequest {
149        messages: vec![system_message, user_message],
150        data_sources: vec![],
151        params,
152        response_format: Some(LLMRequestResponseFormatParam {
153            format_type: JSONType::JsonSchema,
154            json_schema: JSONSchema {
155                name: "CmsParagraphSuggestionResponse".to_string(),
156                strict: true,
157                schema: Schema {
158                    type_field: JSONType::Object,
159                    properties: HashMap::from([(
160                        "suggestions".to_string(),
161                        ArrayProperty {
162                            type_field: JSONType::Array,
163                            items: ArrayItem {
164                                type_field: JSONType::String,
165                            },
166                        },
167                    )]),
168                    required: vec!["suggestions".to_string()],
169                    additional_properties: false,
170                },
171            },
172        }),
173        stop: None,
174    };
175
176    let completion = make_blocking_llm_request(chat_request, app_config, &task_lm).await?;
177
178    let completion_content: &String = &parse_text_completion(completion)?;
179    let response: CmsParagraphSuggestionResponse = serde_json::from_str(completion_content)
180        .map_err(|_| {
181            ChatbotError::new(
182                ChatbotErrorType::ChatbotMessageSuggestError,
183                "The CMS paragraph suggestion LLM returned an incorrectly formatted response."
184                    .to_string(),
185                None,
186            )
187        })?;
188
189    if response.suggestions.is_empty() {
190        return Err(ChatbotError::new(
191            ChatbotErrorType::ChatbotMessageSuggestError,
192            "The CMS paragraph suggestion LLM returned an empty suggestions list.".to_string(),
193            None,
194        ));
195    }
196
197    Ok(response.suggestions)
198}
