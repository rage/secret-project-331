131pub fn fixed_stage_order() -> [CourseDesignerStage; 5] {
132    [
133        CourseDesignerStage::Analysis,
134        CourseDesignerStage::Design,
135        CourseDesignerStage::Development,
136        CourseDesignerStage::Implementation,
137        CourseDesignerStage::Evaluation,
138    ]
139}

141pub fn validate_schedule_input(stages: &[CourseDesignerScheduleStageInput]) -> ModelResult<()> {
142    let expected_order = fixed_stage_order();
143    if stages.len() != expected_order.len() {
144        return Err(ModelError::new(
145            ModelErrorType::InvalidRequest,
146            "Schedule must contain exactly 5 stages.".to_string(),
147            None,
148        ));
149    }
150
151    for (idx, stage) in stages.iter().enumerate() {
152        if stage.stage != expected_order[idx] {
153            return Err(ModelError::new(
154                ModelErrorType::InvalidRequest,
155                "Schedule stages must be in the fixed order: analysis, design, development, implementation, evaluation."
156                    .to_string(),
157                None,
158            ));
159        }
160        if stage.planned_starts_on > stage.planned_ends_on {
161            return Err(ModelError::new(
162                ModelErrorType::InvalidRequest,
163                format!("Stage {:?} starts after it ends.", stage.stage),
164                None,
165            ));
166        }
167        if idx > 0 {
168            let prev = &stages[idx - 1];
169            if !no_gap_between(prev.planned_ends_on, stage.planned_starts_on) {
170                return Err(ModelError::new(
171                    ModelErrorType::InvalidRequest,
172                    "Schedule must have no gaps or overlaps between consecutive stages."
173                        .to_string(),
174                    None,
175                ));
176            }
177        }
178    }
179
180    Ok(())
181}

183fn first_day_of_month(date: NaiveDate) -> ModelResult<NaiveDate> {
184    NaiveDate::from_ymd_opt(date.year(), date.month(), 1).ok_or_else(|| {
185        ModelError::new(
186            ModelErrorType::InvalidRequest,
187            "Invalid date while generating schedule suggestion.".to_string(),
188            None,
189        )
190    })
191}
192
193fn last_day_of_month(year: i32, month: u32) -> ModelResult<u32> {
194    for day in (28..=31).rev() {
195        if NaiveDate::from_ymd_opt(year, month, day).is_some() {
196            return Ok(day);
197        }
198    }
199
200    Err(ModelError::new(
201        ModelErrorType::InvalidRequest,
202        "Invalid month while generating schedule suggestion.".to_string(),
203        None,
204    ))
205}

207pub(crate) fn add_months_clamped(date: NaiveDate, months: u32) -> ModelResult<NaiveDate> {
208    let total_months = date.year() * 12 + date.month0() as i32 + months as i32;
209    let target_year = total_months.div_euclid(12);
210    let target_month0 = total_months.rem_euclid(12) as u32;
211    let target_month = target_month0 + 1;
212    let target_day = date
213        .day()
214        .min(last_day_of_month(target_year, target_month)?);
215
216    NaiveDate::from_ymd_opt(target_year, target_month, target_day).ok_or_else(|| {
217        ModelError::new(
218            ModelErrorType::InvalidRequest,
219            "Failed to generate schedule suggestion date.".to_string(),
220            None,
221        )
222    })
223}

225fn suggestion_months(size: CourseDesignerCourseSize) -> [u32; 5] {
226    match size {
227        CourseDesignerCourseSize::Small => [1, 1, 2, 1, 1],
228        CourseDesignerCourseSize::Medium => [1, 2, 3, 2, 1],
229        CourseDesignerCourseSize::Large => [2, 2, 4, 3, 1],
230    }
231}

233pub fn build_schedule_suggestion(
234    size: CourseDesignerCourseSize,
235    starts_on: NaiveDate,
236) -> ModelResult<Vec<CourseDesignerScheduleStageInput>> {
237    let mut current_start = first_day_of_month(starts_on)?;
238    let stage_order = fixed_stage_order();
239    let month_durations = suggestion_months(size);
240    let mut out = Vec::with_capacity(stage_order.len());
241
242    for (stage, months) in stage_order.into_iter().zip(month_durations) {
243        let next_stage_start = add_months_clamped(current_start, months)?;
244        let planned_ends_on = next_stage_start - Duration::days(1);
245        out.push(CourseDesignerScheduleStageInput {
246            stage,
247            planned_starts_on: current_start,
248            planned_ends_on,
249        });
250        current_start = planned_ends_on + Duration::days(1);
251    }
252
253    Ok(out)
254}

286pub async fn create_plan(
287    conn: &mut PgConnection,
288    user_id: Uuid,
289    name: Option<String>,
290) -> ModelResult<CourseDesignerPlan> {
291    let mut tx = conn.begin().await?;
292
293    let plan: CourseDesignerPlan = sqlx::query_as!(
294        CourseDesignerPlan,
295        r#"
296INSERT INTO course_designer_plans (created_by_user_id, name)
297VALUES ($1, $2)
298RETURNING
299  id,
300  created_at,
301  updated_at,
302  created_by_user_id,
303  name,
304  status AS "status: CourseDesignerPlanStatus",
305  active_stage AS "active_stage: CourseDesignerStage",
306  last_weekly_stage_email_sent_at
307"#,
308        user_id,
309        name
310    )
311    .fetch_one(&mut *tx)
312    .await?;
313
314    sqlx::query!(
315        r#"
316INSERT INTO course_designer_plan_members (course_designer_plan_id, user_id)
317VALUES ($1, $2)
318"#,
319        plan.id,
320        user_id
321    )
322    .execute(&mut *tx)
323    .await?;
324
325    insert_plan_event(
326        &mut tx,
327        plan.id,
328        Some(user_id),
329        "plan_created",
330        None,
331        json!({ "name": plan.name }),
332    )
333    .await?;
334
335    tx.commit().await?;
336    Ok(plan)
337}

339pub async fn get_plan_details_for_user(
340    conn: &mut PgConnection,
341    plan_id: Uuid,
342    user_id: Uuid,
343) -> ModelResult<CourseDesignerPlanDetails> {
344    let plan = course_designer_plan_members::get_plan_for_user(conn, plan_id, user_id).await?;
345    let members =
346        course_designer_plan_members::get_plan_members_for_user(conn, plan_id, user_id).await?;
347    let stages =
348        course_designer_plan_members::get_plan_stages_for_user(conn, plan_id, user_id).await?;
349    let tasks =
350        course_designer_plan_members::get_plan_tasks_for_user(conn, plan_id, user_id).await?;
351    let mut tasks_by_stage: std::collections::HashMap<Uuid, Vec<CourseDesignerPlanStageTask>> =
352        tasks
353            .into_iter()
354            .fold(std::collections::HashMap::new(), |mut acc, t| {
355                acc.entry(t.course_designer_plan_stage_id)
356                    .or_default()
357                    .push(t);
358                acc
359            });
360    let stages_with_tasks: Vec<CourseDesignerPlanStageWithTasks> = stages
361        .into_iter()
362        .map(|stage| {
363            let stage_id = stage.id;
364            let stage_tasks = tasks_by_stage.remove(&stage_id).unwrap_or_default();
365            CourseDesignerPlanStageWithTasks {
366                stage,
367                tasks: stage_tasks,
368            }
369        })
370        .collect();
371    Ok(CourseDesignerPlanDetails {
372        plan,
373        members,
374        stages: stages_with_tasks,
375    })
376}

378pub async fn update_stage_workspace_for_user(
379    conn: &mut PgConnection,
380    plan_id: Uuid,
381    user_id: Uuid,
382    stage: CourseDesignerStage,
383    workspace: crate::course_designer_analysis_workspace::CourseDesignerStageWorkspace,
384) -> ModelResult<CourseDesignerPlanDetails> {
385    if stage != CourseDesignerStage::Analysis {
386        return Err(ModelError::new(
387            ModelErrorType::InvalidRequest,
388            "Workspace payload is only supported for the analysis stage.".to_string(),
389            None,
390        ));
391    }
392    let workspace_json = crate::course_designer_analysis_workspace::workspace_to_json(Some(
393        workspace,
394    ))?
395    .ok_or_else(|| {
396        ModelError::new(
397            ModelErrorType::InvalidRequest,
398            "Workspace serialization produced no data.".to_string(),
399            None,
400        )
401    })?;
402    let updated: Option<Uuid> = sqlx::query_scalar!(
403        r#"
404UPDATE course_designer_plan_stages s
405SET workspace_data = $3
406FROM course_designer_plan_members m
407WHERE s.course_designer_plan_id = $1
408  AND s.stage = $2
409  AND s.deleted_at IS NULL
410  AND m.course_designer_plan_id = s.course_designer_plan_id
411  AND m.user_id = $4
412  AND m.deleted_at IS NULL
413RETURNING s.id
414"#,
415        plan_id,
416        stage as CourseDesignerStage,
417        workspace_json,
418        user_id
419    )
420    .fetch_optional(&mut *conn)
421    .await?;
422    if updated.is_none() {
423        return Err(ModelError::new(
424            ModelErrorType::PreconditionFailed,
425            "Stage not found or user is not a plan member.".to_string(),
426            None,
427        ));
428    }
429    get_plan_details_for_user(conn, plan_id, user_id).await
430}
