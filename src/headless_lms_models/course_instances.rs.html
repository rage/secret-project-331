317pub struct ChapterScore {
318    #[serde(flatten)]
319    pub chapter: DatabaseChapter,
320    pub score_given: f32,
321    pub score_total: i32,
322}

324#[derive(Debug, Default, Serialize)]
325#[cfg_attr(feature = "ts_rs", derive(TS))]
326pub struct PointMap(pub HashMap<Uuid, f32>);
327
328#[derive(Debug, Serialize)]
329#[cfg_attr(feature = "ts_rs", derive(TS))]
330pub struct Points {
331    pub chapter_points: Vec<ChapterScore>,
332    pub users: Vec<UserDetail>,
333    // PointMap is a workaround for https://github.com/rhys-vdw/ts-auto-guard/issues/158
334    pub user_chapter_points: HashMap<Uuid, PointMap>,
335}
336
337pub async fn get_points(
338    conn: &mut PgConnection,
339    instance_id: Uuid,
340    _pagination: Pagination, // TODO
341) -> ModelResult<Points> {
342    let mut chapter_point_totals = HashMap::<Uuid, i32>::new();
343    let mut exercise_to_chapter_id = HashMap::new();
344    let course_instance = crate::course_instances::get_course_instance(conn, instance_id).await?;
345    let exercises =
346        exercises::get_exercises_by_course_id(&mut *conn, course_instance.course_id).await?;
347    for exercise in exercises {
348        if let Some(chapter_id) = exercise.chapter_id {
349            // exercises without chapter ids (i.e. exams) are not counted
350            let total = chapter_point_totals.entry(chapter_id).or_default();
351            *total += exercise.score_maximum;
352            exercise_to_chapter_id.insert(exercise.id, chapter_id);
353        }
354    }
355
356    let users: HashMap<Uuid, User> =
357        users::get_users_by_course_instance_enrollment(conn, instance_id)
358            .await?
359            .into_iter()
360            .map(|u| (u.id, u))
361            .collect();
362    let mut chapter_points_given = HashMap::<Uuid, f32>::new();
363    let states = sqlx::query!(
364        "
365SELECT user_id,
366  exercise_id,
367  score_given
368FROM user_exercise_states
369WHERE course_id = $1
370  AND deleted_at IS NULL
371ORDER BY user_id ASC
372",
373        course_instance.course_id,
374    )
375    .fetch_all(&mut *conn)
376    .await?;
377    let mut user_chapter_points = HashMap::<Uuid, PointMap>::new();
378    for state in states {
379        let user = match users.get(&state.user_id) {
380            Some(user) => user,
381            None => {
382                tracing::warn!(
383                    "user {} has an exercise state but no enrollment",
384                    state.user_id
385                );
386                continue;
387            }
388        };
389        if let Some(chapter_id) = exercise_to_chapter_id.get(&state.exercise_id).copied() {
390            let chapter_points = user_chapter_points.entry(user.id).or_default();
391            let user_given = chapter_points.0.entry(chapter_id).or_default();
392            let chapter_given = chapter_points_given.entry(chapter_id).or_default();
393            let score_given = state.score_given.unwrap_or_default();
394            *user_given += score_given;
395            *chapter_given += score_given;
396        }
397    }
398
399    let chapters = chapters::course_instance_chapters(&mut *conn, instance_id).await?;
400    let mut chapter_points: Vec<ChapterScore> = chapters
401        .into_iter()
402        .map(|c| ChapterScore {
403            score_given: chapter_points_given.get(&c.id).copied().unwrap_or_default(),
404            score_total: chapter_point_totals.get(&c.id).copied().unwrap_or_default(),
405            chapter: c,
406        })
407        .collect();
408    chapter_points.sort_by_key(|c| c.chapter.chapter_number);
409
410    let list_of_users = users.into_values().collect::<Vec<_>>();
411    let user_id_to_details =
412        crate::user_details::get_users_details_by_user_id_map(&mut *conn, &list_of_users).await?;
413
414    Ok(Points {
415        chapter_points,
416        users: list_of_users
417            .into_iter()
418            .filter_map(|user| user_id_to_details.get(&user.id).cloned())
419            .collect::<Vec<_>>(),
420        user_chapter_points,
421    })
422}

483pub async fn is_open(conn: &mut PgConnection, id: Uuid) -> ModelResult<bool> {
484    let res = sqlx::query!(
485        "
486SELECT starts_at,
487  ends_at
488FROM course_instances
489WHERE id = $1
490",
491        id
492    )
493    .fetch_one(conn)
494    .await?;
495    let has_started = match res.starts_at {
496        Some(starts_at) => starts_at <= Utc::now(),
497        None => true,
498    };
499    let has_ended = match res.ends_at {
500        Some(ends_at) => ends_at <= Utc::now(),
501        None => false,
502    };
503    let is_open = has_started && !has_ended;
504    Ok(is_open)
505}

782        user_id,
783        course_id
784    )
785    .execute(&mut *tx)
786    .await?;
787    sqlx::query!(
788        "
789UPDATE generated_certificates
790SET deleted_at = NOW()
791WHERE user_id = $1
792  AND certificate_configuration_id IN (
793    SELECT certificate_configuration_id
794    FROM certificate_configuration_to_requirements
795    WHERE course_module_id IN (
796        SELECT id
797        FROM course_modules
798        WHERE course_id = $2
799      )
800      AND deleted_at IS NULL
801  )
802  AND deleted_at IS NULL
803",
804        user_id,
805        course_id
806    )
807    .execute(&mut *tx)
808    .await?;
809
810    tx.commit().await?;
811    Ok(())

814pub async fn get_course_average_duration(
815    conn: &mut PgConnection,
816    course_id: Uuid,
817) -> ModelResult<Option<i64>> {
818    let res = sqlx::query!(
819        "
820SELECT AVG(
821    EXTRACT(
822      EPOCH
823      FROM cmc.completion_date - ce.created_at
824    )
825  )::int8 AS average_duration_seconds
826FROM course_instance_enrollments ce
827  JOIN course_module_completions cmc ON (
828    cmc.course_id = ce.course_id
829    AND cmc.user_id = ce.user_id
830  )
831WHERE ce.course_id = $1
832  AND ce.deleted_at IS NULL
833  AND cmc.deleted_at IS NULL;
834        ",
835        course_id
836    )
837    .fetch_optional(conn)
838    .await?;
839
840    Ok(res.map(|r| r.average_duration_seconds).unwrap_or_default())
841}

843pub async fn get_student_duration(
844    conn: &mut PgConnection,
845    user_id: Uuid,
846    course_id: Uuid,
847) -> ModelResult<Option<i64>> {
848    let res = sqlx::query!(
849        "
850SELECT COALESCE(
851    EXTRACT(
852      EPOCH
853      FROM cmc.completion_date - ce.created_at
854    )::int8,
855    0
856  ) AS student_duration_seconds
857FROM course_instance_enrollments ce
858  JOIN course_module_completions cmc ON (
859    cmc.course_id = ce.course_id
860    AND cmc.user_id = ce.user_id
861  )
862WHERE ce.course_id = $1
863  AND ce.user_id = $2
864  AND ce.deleted_at IS NULL
865  AND cmc.deleted_at IS NULL;
866        ",
867        course_id,
868        user_id
869    )
870    .fetch_optional(conn)
871    .await?;
872
873    Ok(res.map(|r| r.student_duration_seconds).unwrap_or_default())
874}
