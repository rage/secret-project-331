528/// Completion in the form that is recognized by authorized third party study registry registrars.
529#[derive(Clone, PartialEq, Deserialize, Serialize)]
530#[cfg_attr(feature = "ts_rs", derive(TS))]
531pub struct StudyRegistryCompletion {
532    /// The date when the student completed the course. The value of this field is the date that will
533    /// end up in the user's study registry as the completion date. If the completion is created
534    /// automatically, it is the date when the student passed the completion thresholds. If the teacher
535    /// creates these completions manually, the teacher inputs this value. Usually the teacher would in
536    /// this case input the date of the exam.
537    pub completion_date: DateTime<Utc>,
538    /// The language used in the completion of the course.
539    pub completion_language: String,
540    /// Date when the student opened the form to register their credits to the open university.
541    pub completion_registration_attempt_date: Option<DateTime<Utc>>,
542    /// Email at the time of completing the course. Used to match the student to the data that they will
543    /// fill to the open university and it will remain unchanged in the event of email change because
544    /// changing this would break the matching.
545    pub email: String,
546    /// The grade to be passed to the study registry. Uses the sisu format. See the struct documentation for details.
547    pub grade: StudyRegistryGrade,
548    /// ID of the completion.
549    pub id: Uuid,
550    /// User id in courses.mooc.fi for received registered completions.
551    pub user_id: Uuid,
552    /// Tier of the completion. Currently always null. Historically used for example to distinguish between
553    /// intermediate and advanced versions of the Building AI course.
554    pub tier: Option<i32>,
555}

557impl From<CourseModuleCompletion> for StudyRegistryCompletion {
558    fn from(completion: CourseModuleCompletion) -> Self {
559        Self {
560            completion_date: completion.completion_date,
561            completion_language: completion.completion_language,
562            completion_registration_attempt_date: completion.completion_registration_attempt_date,
563            email: completion.email,
564            grade: StudyRegistryGrade::new(completion.passed, completion.grade),
565            id: completion.id,

569    }

572/// Grading object that maps the system grading information to Sisu's grading scales.
573///
574/// Currently only `sis-0-5` and `sis-hyv-hyl` scales are supported in the system.
575///
576/// All grading scales can be found from <https://sis-helsinki-test.funidata.fi/api/graphql> using
577/// the following query:
578///
579/// ```graphql
580/// query {
581///   grade_scales {
582///     id
583///     name {
584///       fi
585///       en
586///       sv
587///     }
588///     grades {
589///       name {
590///         fi
591///         en
592///         sv
593///       }
594///       passed
595///       localId
596///       abbreviation {
597///         fi
598///         en
599///         sv
600///       }
601///     }
602///     abbreviation {
603///       fi
604///       en
605///       sv
606///     }
607///   }
608/// }
609/// ```
610#[derive(Clone, PartialEq, Deserialize, Serialize)]
611#[cfg_attr(feature = "ts_rs", derive(TS))]
612pub struct StudyRegistryGrade {
613    pub scale: String,
614    pub grade: String,
615}
616
617impl StudyRegistryGrade {
618    pub fn new(passed: bool, grade: Option<i32>) -> Self {
619        match grade {
620            Some(grade) => Self {
621                scale: "sis-0-5".to_string(),
622                grade: grade.to_string(),
623            },
624            None => Self {
625                scale: "sis-hyv-hyl".to_string(),
626                grade: if passed {
627                    "1".to_string()
628                } else {
629                    "0".to_string()
630                },
631            },
632        }
633    }
634}
635/// Streams completions.
636///
637/// If no_completions_registered_by_this_study_registry_registrar is None, then all completions are streamed.
638pub fn stream_by_course_module_id<'a>(
639    conn: &'a mut PgConnection,
640    course_module_ids: &'a [Uuid],
641    no_completions_registered_by_this_study_registry_registrar: &'a Option<StudyRegistryRegistrar>,
642) -> impl Stream<Item = sqlx::Result<StudyRegistryCompletion>> + Send + 'a {
643    // If this is none, we're using a null uuid, which will never match anything. Therefore, no completions will be filtered out.
644    let study_module_registrar_id = no_completions_registered_by_this_study_registry_registrar
645        .clone()
646        .map(|o| o.id)
647        .unwrap_or(Uuid::nil());
648
649    sqlx::query_as!(
650        CourseModuleCompletion,
651        r#"
652SELECT *
653FROM course_module_completions
654WHERE course_module_id = ANY($1)
655  AND prerequisite_modules_completed
656  AND eligible_for_ects IS TRUE
657  AND deleted_at IS NULL
658  AND id NOT IN (
659    SELECT course_module_completion_id
660    FROM course_module_completion_registered_to_study_registries
661    WHERE course_module_id = ANY($1)
662      AND study_registry_registrar_id = $2
663      AND deleted_at IS NULL
664  )
665        "#,
666        course_module_ids,
667        study_module_registrar_id,
668    )
669    .map(StudyRegistryCompletion::from)
670    .fetch(conn)
671}
