78) -> ModelResult<Uuid> {
79    let res = sqlx::query!(

81INSERT INTO exercise_slides (id, exercise_id, order_number)

85    order_number = $3,

87RETURNING id;
88    ",
89        id,

220pub async fn delete_exercise_slides_by_exercise_ids(
