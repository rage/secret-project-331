125/**
126Helper struct to use with functions that insert data into the database.
127
128## Examples
129
130### Usage when inserting to a database
131
132By calling `.into_uuid()` function implemented by `PKeyPolicy<Uuid>`, this enum can be used with
133SQLX queries while letting the caller dictate how the primary key should be decided.
134
135```no_check
136# use headless_lms_models::{ModelResult, PKeyPolicy};
137# use uuid::Uuid;
138# use sqlx::PgConnection;
139async fn insert(
140    conn: &mut PgConnection,
141    pkey_policy: PKeyPolicy<Uuid>,
142) -> ModelResult<Uuid> {
143    let res = sqlx::query!(
144        "INSERT INTO organizations (id) VALUES ($1) RETURNING id",
145        pkey_policy.into_uuid(),
146    )
147    .fetch_one(conn)
148    .await?;
149    Ok(res.id)
150}
151
152# async fn random_function(conn: &mut PgConnection) -> ModelResult<()> {
153// Insert using generated id.
154let foo_1_id = insert(conn, PKeyPolicy::Generate).await.unwrap();
155
156// Insert using fixed id.
157let uuid = Uuid::parse_str("8fce44cf-738e-4fc9-8d8e-47c350fd3a7f").unwrap();
158let foo_2_id = insert(conn, PKeyPolicy::Fixed(uuid)).await.unwrap();
159assert_eq!(foo_2_id, uuid);
160# Ok(())
161# }
162```
163
164### Usage in a higher-order function.
165
166When `PKeyPolicy` is used with a higher-order function, an arbitrary struct can be provided
167instead. The data can be mapped further by calling the `.map()` or `.map_ref()` methods.
168
169```no_run
170# use headless_lms_models::{ModelResult, PKeyPolicy};
171# use uuid::Uuid;
172# use sqlx::PgConnection;
173# mod foos {
174#   use headless_lms_models::{ModelResult, PKeyPolicy};
175#   use uuid::Uuid;
176#   use sqlx::PgConnection;
177#   pub async fn insert(conn: &mut PgConnection, pkey_policy: PKeyPolicy<Uuid>) -> ModelResult<()> {
178#       Ok(())
179#   }
180# }
181# mod bars {
182#   use headless_lms_models::{ModelResult, PKeyPolicy};
183#   use uuid::Uuid;
184#   use sqlx::PgConnection;
185#   pub async fn insert(conn: &mut PgConnection, pkey_policy: PKeyPolicy<Uuid>) -> ModelResult<()> {
186#       Ok(())
187#   }
188# }
189
190struct FooBar {
191    foo: Uuid,
192    bar: Uuid,
193}
194
195async fn multiple_inserts(
196    conn: &mut PgConnection,
197    pkey_policy: PKeyPolicy<FooBar>,
198) -> ModelResult<()> {
199    foos::insert(conn, pkey_policy.map_ref(|x| x.foo)).await?;
200    bars::insert(conn, pkey_policy.map_ref(|x| x.bar)).await?;
201    Ok(())
202}
203
204# async fn some_function(conn: &mut PgConnection) {
205// Insert using generated ids.
206assert!(multiple_inserts(conn, PKeyPolicy::Generate).await.is_ok());
207
208// Insert using fixed ids.
209let foobar = FooBar {
210    foo: Uuid::parse_str("52760668-cc9d-4144-9226-d2aacb83bea9").unwrap(),
211    bar: Uuid::parse_str("ce9bd0cd-0e66-4522-a1b4-52a9347a115c").unwrap(),
212};
213assert!(multiple_inserts(conn, PKeyPolicy::Fixed(foobar)).await.is_ok());
214# }
215```
216*/
217pub enum PKeyPolicy<T> {
218    /// Ids will be generated based on the associated data. Usually only used in
219    /// local test environments where reproducible database states are desired.
220    Fixed(T),
221    /// Ids will be generated on the database level. This should be the default
222    /// behavior.
223    Generate,
224}
225
226impl<T> PKeyPolicy<T> {
227    /// Gets reference to the fixed data, if there are any.
228    pub fn fixed(&self) -> Option<&T> {
229        match self {
230            PKeyPolicy::Fixed(t) => Some(t),
231            PKeyPolicy::Generate => None,
232        }
233    }
234
235    /// Maps `PKeyPolicy<T>` to `PKeyPolicy<U>` by applying a function to the contained value.
236    pub fn map<U, F>(self, f: F) -> PKeyPolicy<U>
237    where
238        F: FnOnce(T) -> U,
239    {
240        match self {
241            PKeyPolicy::Fixed(x) => PKeyPolicy::Fixed(f(x)),
242            PKeyPolicy::Generate => PKeyPolicy::Generate,
243        }
244    }
245
246    /// Maps a reference of contained data in `Fixed(T)` to `PKeyPolicy<U>` by applying a function
247    /// to the contained value. This is useful whenever a referenced value can be used instead of
248    /// having to consume the original value.
249    pub fn map_ref<U, F>(&self, f: F) -> PKeyPolicy<U>
250    where
251        F: FnOnce(&T) -> U,
252    {
253        match self {
254            PKeyPolicy::Fixed(x) => PKeyPolicy::Fixed(f(x)),
255            PKeyPolicy::Generate => PKeyPolicy::Generate,
256        }
257    }
258}
259
260impl PKeyPolicy<Uuid> {
261    /// Maps into the contained `Uuid` value or generates a new one.
262    pub fn into_uuid(self) -> Uuid {
263        match self {
264            PKeyPolicy::Fixed(uuid) => uuid,
265            PKeyPolicy::Generate => Uuid::new_v4(),
266        }
267    }
268}

294/// Either a course or exam id.
295///
296/// Exercises can either be part of courses or exams. Many user-related actions need to differentiate
297/// between two, so `CourseOrExamId` helps when handling these separate scenarios.
298#[derive(Debug, Serialize, Deserialize, PartialEq, Eq, Clone, Copy, Hash)]
299pub enum CourseOrExamId {
300    Course(Uuid),
301    Exam(Uuid),
302}
303
304impl CourseOrExamId {
305    pub fn from_course_and_exam_ids(
306        course_id: Option<Uuid>,
307        exam_id: Option<Uuid>,
308    ) -> ModelResult<Self> {
309        match (course_id, exam_id) {
310            (None, None) => Err(ModelError::new(
311                ModelErrorType::Generic,
312                "Expected either course or exam id, but neither were provided.",
313                None,
314            )),
315            (Some(course_id), None) => Ok(Self::Course(course_id)),
316            (None, Some(exam_id)) => Ok(Self::Exam(exam_id)),
317            (Some(_), Some(_)) => Err(ModelError::new(
318                ModelErrorType::Generic,
319                "Expected either course or exam id, but both were provided.",
320                None,
321            )),
322        }
323    }
324
325    pub fn to_course_and_exam_ids(&self) -> (Option<Uuid>, Option<Uuid>) {
326        match self {
327            CourseOrExamId::Course(course_id) => (Some(*course_id), None),
328            CourseOrExamId::Exam(exam_id) => (None, Some(*exam_id)),
329        }
330    }
331}
