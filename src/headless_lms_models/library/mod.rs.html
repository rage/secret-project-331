16#[derive(Debug, Clone, Copy, PartialEq, Eq, Serialize, Deserialize)]
17#[cfg_attr(feature = "ts_rs", derive(TS))]
18pub enum TimeGranularity {
19    Year,
20    Month,
21    Day,
22}
23
24impl TimeGranularity {
25    /// Returns (interval_unit, time_unit) for use in SQL queries
26    /// e.g., ("years", "year") for TimeGranularity::Year
27    pub fn get_sql_units(&self) -> (&'static str, &'static str) {
28        match self {
29            TimeGranularity::Year => ("years", "year"),
30            TimeGranularity::Month => ("months", "month"),
31            TimeGranularity::Day => ("days", "day"),
32        }
33    }
34}
35
36impl std::fmt::Display for TimeGranularity {
37    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
38        match self {
39            TimeGranularity::Year => write!(f, "Year"),
40            TimeGranularity::Month => write!(f, "Month"),
41            TimeGranularity::Day => write!(f, "Day"),
42        }
43    }
44}
45
46#[derive(Debug)]
47pub struct ParseTimeGranularityError;
48
49impl std::fmt::Display for ParseTimeGranularityError {
50    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
51        write!(f, "invalid time granularity")
52    }
53}
54
55impl std::error::Error for ParseTimeGranularityError {}
56
57impl std::str::FromStr for TimeGranularity {
58    type Err = ParseTimeGranularityError;
59
60    fn from_str(s: &str) -> Result<Self, Self::Err> {
61        match s.to_lowercase().as_str() {
62            "year" => Ok(TimeGranularity::Year),
63            "month" => Ok(TimeGranularity::Month),
64            "day" => Ok(TimeGranularity::Day),
65            _ => Err(ParseTimeGranularityError),
66        }
67    }
68}

70impl TryFrom<String> for TimeGranularity {
71    type Error = ParseTimeGranularityError;

74        value.parse()
