18pub async fn insert_rejected_exercise_slide_submission(

20    rejected_submission: &StudentExerciseSlideSubmission,
21    user_id: Uuid,
22    http_status_code: Option<i32>,
23    error_message: Option<String>,
24) -> ModelResult<Uuid> {
25    let mut tx = conn.begin().await?;
26    let res = sqlx::query!(
27        "
28INSERT INTO rejected_exercise_slide_submissions (
29    user_id,
30    exercise_slide_id,
31    http_status_code,
32    error_message
33  )
34VALUES ($1, $2, $3, $4)
35RETURNING id
36        ",
37        user_id,
38        rejected_submission.exercise_slide_id,
39        http_status_code,
40        error_message,
41    )
42    .fetch_one(&mut *tx)
43    .await?;
44
45    for task in &rejected_submission.exercise_task_submissions {
46        insert_rejected_exercise_task_submission(&mut tx, task, res.id).await?;
47    }
48
49    tx.commit().await?;
50    Ok(res.id)
51}
