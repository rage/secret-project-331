20/// Apply length to input number and write to result using decimal_formatter.
21fn try_write_number<W>(
22    part: Part,
23    w: &mut W,
24    decimal_formatter: Option<&DecimalFormatter>,
25    mut num: Decimal,
26    length: FieldLength,
27) -> Result<Result<(), FormattedDateTimePatternError>, fmt::Error>
28where
29    W: writeable::PartsWrite + ?Sized,
30{
31    num.pad_start(length.to_len() as i16);
32
33    if let Some(fdf) = decimal_formatter {
34        w.with_part(part, |w| fdf.format(&num).write_to_parts(w))?;
35        Ok(Ok(()))
36    } else {
37        w.with_part(part, |w| {
38            w.with_part(writeable::Part::ERROR, |r| num.write_to_parts(r))
39        })?;
40        Ok(Err(
41            FormattedDateTimePatternError::DecimalFormatterNotLoaded,
42        ))
43    }
44}

46/// Apply length to input number and write to result using decimal_formatter.
47/// Don't annotate it with a part.
48fn try_write_number_without_part<W>(
49    w: &mut W,
50    decimal_formatter: Option<&DecimalFormatter>,
51    mut num: Decimal,
52    length: FieldLength,
53) -> Result<Result<(), FormattedDateTimePatternError>, fmt::Error>
54where
55    W: writeable::PartsWrite + ?Sized,
56{
57    num.pad_start(length.to_len() as i16);
58
59    if let Some(fdf) = decimal_formatter {
60        fdf.format(&num).write_to(w)?;
61        Ok(Ok(()))
62    } else {
63        w.with_part(writeable::Part::ERROR, |r| num.write_to(r))?;
64        Ok(Err(
65            FormattedDateTimePatternError::DecimalFormatterNotLoaded,
66        ))
67    }
68}

70pub(crate) fn try_write_pattern_items<W>(
71    pattern_metadata: PatternMetadata,
72    pattern_items: impl Iterator<Item = PatternItem>,
73    input: &DateTimeInputUnchecked,
74    datetime_names: &RawDateTimeNamesBorrowed,
75    decimal_formatter: Option<&DecimalFormatter>,
76    w: &mut W,
77) -> Result<Result<(), FormattedDateTimePatternError>, fmt::Error>
78where
79    W: writeable::PartsWrite + ?Sized,
80{
81    let mut r = Ok(());
82    for item in pattern_items {
83        match item {
84            PatternItem::Literal(ch) => w.write_char(ch)?,
85            PatternItem::Field(field) => {
86                r = r.and(try_write_field(
87                    field,
88                    pattern_metadata,
89                    input,
90                    datetime_names,
91                    decimal_formatter,
92                    w,
93                )?);
94            }
95        }
96    }
97    Ok(r)
98}

100// This function assumes that the correct decision has been
101// made regarding availability of symbols in the caller.
102//
103// When modifying the list of fields using symbols,
104// update the matching query in `analyze_pattern` function.
105fn try_write_field<W>(
106    field: fields::Field,
107    pattern_metadata: PatternMetadata,
108    input: &DateTimeInputUnchecked,
109    datetime_names: &RawDateTimeNamesBorrowed,
110    decimal_formatter: Option<&DecimalFormatter>,
111    w: &mut W,
112) -> Result<Result<(), FormattedDateTimePatternError>, fmt::Error>
113where
114    W: writeable::PartsWrite + ?Sized,
115{
116    macro_rules! input {
117        // Get the input. If not found, write a replacement string but do NOT write a part.
118        (_, $kind:ident, $name:ident = $input:expr) => {
119            let Some($name) = $input else {
120                write_value_missing(w, field)?;
121                return Ok(Err(FormattedDateTimePatternError::MissingInputField(
122                    MissingInputFieldKind::$kind,
123                )));
124            };
125        };
126        // Get the input. If not found, write a replacement string and a part.
127        ($part:ident, $kind:ident, $name:ident = $input:expr) => {
128            let Some($name) = $input else {
129                w.with_part($part, |w| write_value_missing(w, field))?;
130                return Ok(Err(FormattedDateTimePatternError::MissingInputField(
131                    MissingInputFieldKind::$kind,
132                )));
133            };
134        };
135    }
136
137    Ok(match (field.symbol, field.length) {
138        (FieldSymbol::Era, l) => {
139            const PART: Part = parts::ERA;
140            input!(PART, Year, year = input.year);
141            input!(PART, YearEra, era_year = year.era());
142            let era_symbol = datetime_names
143                .get_name_for_era(l, era_year)
144                .map_err(|e| match e {
145                    GetNameForEraError::InvalidEraCode => {
146                        FormattedDateTimePatternError::InvalidEra(era_year.era)
147                    }
148                    GetNameForEraError::InvalidFieldLength => {
149                        FormattedDateTimePatternError::UnsupportedLength(ErrorField(field))
150                    }
151                    GetNameForEraError::NotLoaded => {
152                        FormattedDateTimePatternError::NamesNotLoaded(ErrorField(field))
153                    }
154                });
155            match era_symbol {
156                Err(e) => {
157                    w.with_part(PART, |w| {
158                        w.with_part(Part::ERROR, |w| w.write_str(&era_year.era))
159                    })?;
160                    Err(e)
161                }
162                Ok(era) => Ok(w.with_part(PART, |w| w.write_str(era))?),
163            }
164        }
165        (FieldSymbol::Year(Year::Calendar), l) => {
166            const PART: Part = parts::YEAR;
167            input!(PART, Year, year = input.year);
168            let mut year = Decimal::from(year.era_year_or_related_iso());
169            if matches!(l, FieldLength::Two) {
170                // 'yy' and 'YY' truncate
171                year.set_max_position(2);
172            }
173            try_write_number(PART, w, decimal_formatter, year, l)?
174        }
175        (FieldSymbol::Year(Year::Cyclic), l) => {
176            const PART: Part = parts::YEAR_NAME;
177            input!(PART, Year, year = input.year);
178            input!(PART, YearCyclic, cyclic = year.cyclic());
179
180            match datetime_names.get_name_for_cyclic(l, cyclic.year) {
181                Ok(name) => Ok(w.with_part(PART, |w| w.write_str(name))?),
182                Err(e) => {
183                    w.with_part(PART, |w| {
184                        w.with_part(Part::ERROR, |w| {
185                            try_write_number_without_part(
186                                w,
187                                decimal_formatter,
188                                cyclic.related_iso.into(),
189                                FieldLength::One,
190                            )
191                            .map(|_| ())
192                        })
193                    })?;
194                    return Ok(Err(match e {
195                        GetNameForCyclicYearError::InvalidYearNumber { max } => {
196                            FormattedDateTimePatternError::InvalidCyclicYear {
197                                value: cyclic.year,
198                                max,
199                            }
200                        }
201                        GetNameForCyclicYearError::InvalidFieldLength => {
202                            FormattedDateTimePatternError::UnsupportedLength(ErrorField(field))
203                        }
204                        GetNameForCyclicYearError::NotLoaded => {
205                            FormattedDateTimePatternError::NamesNotLoaded(ErrorField(field))
206                        }
207                    }));
208                }
209            }
210        }
211        (FieldSymbol::Year(Year::RelatedIso), l) => {
212            const PART: Part = parts::RELATED_YEAR;
213            input!(PART, Year, year = input.year);
214            input!(PART, YearCyclic, cyclic = year.cyclic());
215
216            // Always in latin digits according to spec
217            w.with_part(PART, |w| {
218                let mut num = Decimal::from(cyclic.related_iso);
219                num.pad_start(l.to_len() as i16);
220                num.write_to(w)
221            })?;
222            Ok(())
223        }
224        (FieldSymbol::Year(Year::Extended), l) => {
225            const PART: Part = parts::EXTENDED_YEAR;
226            input!(PART, Year, year = input.year);
227            let extended = year.extended_year();
228            try_write_number(PART, w, decimal_formatter, extended.into(), l)?
229        }
230        (FieldSymbol::Month(_), l @ (FieldLength::One | FieldLength::Two)) => {
231            const PART: Part = parts::MONTH;
232            input!(PART, Month, month = input.month);
233            try_write_number(PART, w, decimal_formatter, month.ordinal.into(), l)?
234        }
235        (FieldSymbol::Month(symbol), l) => {
236            const PART: Part = parts::MONTH;
237            input!(PART, Month, month = input.month);
238            match datetime_names.get_name_for_month(
239                symbol,
240                l,
241                month.formatting_month_number() - 1,
242                month.formatting_is_leap(),
243            ) {
244                Ok(MonthPlaceholderValue::PlainString(symbol)) => {
245                    w.with_part(PART, |w| w.write_str(symbol))?;
246                    Ok(())
247                }
248                Ok(MonthPlaceholderValue::Numeric) => {
249                    debug_assert!(l == FieldLength::One);
250                    try_write_number(PART, w, decimal_formatter, month.ordinal.into(), l)?
251                }
252                Ok(MonthPlaceholderValue::NumericPattern(substitution_pattern)) => {
253                    debug_assert!(l == FieldLength::One);
254                    if let Some(formatter) = decimal_formatter {
255                        let mut num = Decimal::from(month.ordinal);
256                        num.pad_start(l.to_len() as i16);
257                        w.with_part(PART, |w| {
258                            substitution_pattern
259                                .interpolate([formatter.format(&num)])
260                                .write_to(w)
261                        })?;
262                        Ok(())
263                    } else {
264                        w.with_part(PART, |w| {
265                            w.with_part(Part::ERROR, |w| {
266                                substitution_pattern
267                                    .interpolate([{
268                                        let mut num = Decimal::from(month.ordinal);
269                                        num.pad_start(l.to_len() as i16);
270                                        num
271                                    }])
272                                    .write_to(w)
273                            })
274                        })?;
275                        Err(FormattedDateTimePatternError::DecimalFormatterNotLoaded)
276                    }
277                }
278                Err(e) => {
279                    w.with_part(PART, |w| {
280                        w.with_part(Part::ERROR, |w| w.write_str(&month.formatting_code.0))
281                    })?;
282                    Err(match e {
283                        GetNameForMonthError::InvalidMonthCode => {
284                            FormattedDateTimePatternError::InvalidMonthCode(month.formatting_code)
285                        }
286                        GetNameForMonthError::InvalidFieldLength => {
287                            FormattedDateTimePatternError::UnsupportedLength(ErrorField(field))
288                        }
289                        GetNameForMonthError::NotLoaded => {
290                            FormattedDateTimePatternError::NamesNotLoaded(ErrorField(field))
291                        }
292                    })
293                }
294            }
295        }
296        (FieldSymbol::Week(w), _) => match w {},
297        (FieldSymbol::Weekday(weekday), l) => {
298            const PART: Part = parts::WEEKDAY;
299            input!(PART, Weekday, iso_weekday = input.iso_weekday);
300            match datetime_names
301                .get_name_for_weekday(weekday, l, iso_weekday)
302                .map_err(|e| match e {
303                    GetNameForWeekdayError::InvalidFieldLength => {
304                        FormattedDateTimePatternError::UnsupportedLength(ErrorField(field))
305                    }
306                    GetNameForWeekdayError::NotLoaded => {
307                        FormattedDateTimePatternError::NamesNotLoaded(ErrorField(field))
308                    }
309                }) {
310                Err(e) => {
311                    w.with_part(PART, |w| {
312                        w.with_part(Part::ERROR, |w| {
313                            w.write_str(match iso_weekday {
314                                Weekday::Monday => "mon",
315                                Weekday::Tuesday => "tue",
316                                Weekday::Wednesday => "wed",
317                                Weekday::Thursday => "thu",
318                                Weekday::Friday => "fri",
319                                Weekday::Saturday => "sat",
320                                Weekday::Sunday => "sun",
321                            })
322                        })
323                    })?;
324                    Err(e)
325                }
326                Ok(s) => Ok(w.with_part(PART, |w| w.write_str(s))?),
327            }
328        }
329        (FieldSymbol::Day(fields::Day::DayOfMonth), l) => {
330            const PART: Part = parts::DAY;
331            input!(PART, DayOfMonth, day_of_month = input.day_of_month);
332            try_write_number(PART, w, decimal_formatter, day_of_month.0.into(), l)?
333        }
334        (FieldSymbol::Day(fields::Day::DayOfWeekInMonth), l) => {
335            input!(_, DayOfMonth, day_of_month = input.day_of_month);
336            try_write_number_without_part(
337                w,
338                decimal_formatter,
339                DayOfWeekInMonth::from(day_of_month).0.into(),
340                l,
341            )?
342        }
343        (FieldSymbol::Day(fields::Day::DayOfYear), l) => {
344            input!(_, DayOfYear, day_of_year = input.day_of_year);
345            try_write_number_without_part(w, decimal_formatter, day_of_year.0.into(), l)?
346        }
347        (FieldSymbol::Day(fields::Day::ModifiedJulianDay), l) => {
348            const PART: Part = parts::JULIAN_DAY;
349            const MODIFIED_JULIAN_DAY_EPOCH: RataDie = RataDie::new(-1721425);
350            input!(_, RataDie, rata_die = input.rata_die);
351            let julian_day = rata_die - MODIFIED_JULIAN_DAY_EPOCH;
352            try_write_number(PART, w, decimal_formatter, julian_day.into(), l)?
353        }
354        (FieldSymbol::Hour(symbol), l) => {
355            const PART: Part = parts::HOUR;
356            input!(PART, Hour, hour = input.hour);
357            let h = hour.number();
358            let h = match symbol {
359                fields::Hour::H11 => h % 12,
360                fields::Hour::H12 => {
361                    let v = h % 12;
362                    if v == 0 {
363                        12
364                    } else {
365                        v
366                    }
367                }
368                fields::Hour::H23 => h,
369            };
370            try_write_number(PART, w, decimal_formatter, h.into(), l)?
371        }
372        (FieldSymbol::Minute, l) => {
373            const PART: Part = parts::MINUTE;
374            input!(PART, Minute, minute = input.minute);
375            try_write_number(PART, w, decimal_formatter, minute.number().into(), l)?
376        }
377        (FieldSymbol::Second(Second::Second), l) => {
378            const PART: Part = parts::SECOND;
379            input!(PART, Second, second = input.second);
380            try_write_number(PART, w, decimal_formatter, second.number().into(), l)?
381        }
382        (FieldSymbol::Second(Second::MillisInDay), l) => {
383            input!(_, Hour, hour = input.hour);
384            input!(_, Minute, minute = input.minute);
385            input!(_, Second, second = input.second);
386            input!(_, Subsecond, subsecond = input.subsecond);
387
388            let milliseconds = (((hour.number() as u32 * 60) + minute.number() as u32) * 60
389                + second.number() as u32)
390                * 1000
391                + subsecond.number() / 1_000_000;
392            try_write_number_without_part(w, decimal_formatter, milliseconds.into(), l)?
393        }
394        (FieldSymbol::DecimalSecond(decimal_second), l) => {
395            const PART: Part = parts::SECOND;
396            input!(PART, Second, second = input.second);
397            input!(PART, Subsecond, subsecond = input.subsecond);
398
399            // Formatting with fractional seconds
400            let mut s = Decimal::from(second.number());
401            let _infallible = s.concatenate_end(
402                Decimal::from(subsecond.number())
403                    .absolute
404                    .multiplied_pow10(-9),
405            );
406            debug_assert!(_infallible.is_ok());
407            let position = -(decimal_second as i16);
408            s.trunc(position);
409            s.pad_end(position);
410            try_write_number(PART, w, decimal_formatter, s, l)?
411        }
412        (FieldSymbol::DayPeriod(period), l) => {
413            const PART: Part = parts::DAY_PERIOD;
414            input!(PART, Hour, hour = input.hour);
415
416            match datetime_names.get_name_for_day_period(
417                period,
418                l,
419                hour,
420                pattern_metadata.time_granularity().is_top_of_hour(
421                    input.minute.unwrap_or_default().number(),
422                    input.second.unwrap_or_default().number(),
423                    input.subsecond.unwrap_or_default().number(),
424                ),
425            ) {
426                Err(e) => {
427                    w.with_part(PART, |w| {
428                        w.with_part(Part::ERROR, |w| {
429                            w.write_str(if hour.number() < 12 { "AM" } else { "PM" })
430                        })
431                    })?;
432                    Err(match e {
433                        GetNameForDayPeriodError::InvalidFieldLength => {
434                            FormattedDateTimePatternError::UnsupportedLength(ErrorField(field))
435                        }
436                        GetNameForDayPeriodError::NotLoaded => {
437                            FormattedDateTimePatternError::NamesNotLoaded(ErrorField(field))
438                        }
439                    })
440                }
441                Ok(s) => Ok(w.with_part(PART, |w| w.write_str(s))?),
442            }
443        }
444        (FieldSymbol::TimeZone(fields::TimeZone::SpecificNonLocation), FieldLength::Four) => {
445            perform_timezone_fallback(
446                w,
447                input,
448                datetime_names,
449                decimal_formatter,
450                field,
451                &[
452                    TimeZoneFormatterUnit::SpecificNonLocation(FieldLength::Four),
453                    TimeZoneFormatterUnit::LocalizedOffset(FieldLength::Four),
454                ],
455            )?
456        }
457        (FieldSymbol::TimeZone(fields::TimeZone::SpecificNonLocation), l) => {
458            perform_timezone_fallback(
459                w,
460                input,
461                datetime_names,
462                decimal_formatter,
463                field,
464                &[
465                    TimeZoneFormatterUnit::SpecificNonLocation(l),
466                    TimeZoneFormatterUnit::LocalizedOffset(l),
467                ],
468            )?
469        }
470        (FieldSymbol::TimeZone(fields::TimeZone::GenericNonLocation), l) => {
471            perform_timezone_fallback(
472                w,
473                input,
474                datetime_names,
475                decimal_formatter,
476                field,
477                &[
478                    TimeZoneFormatterUnit::GenericNonLocation(l),
479                    TimeZoneFormatterUnit::GenericLocation,
480                    TimeZoneFormatterUnit::LocalizedOffset(l),
481                ],
482            )?
483        }
484        (FieldSymbol::TimeZone(fields::TimeZone::Location), FieldLength::Four) => {
485            perform_timezone_fallback(
486                w,
487                input,
488                datetime_names,
489                decimal_formatter,
490                field,
491                &[
492                    TimeZoneFormatterUnit::GenericLocation,
493                    TimeZoneFormatterUnit::LocalizedOffset(FieldLength::Four),
494                ],
495            )?
496        }
497        (FieldSymbol::TimeZone(fields::TimeZone::Location), FieldLength::Three) => {
498            perform_timezone_fallback(
499                w,
500                input,
501                datetime_names,
502                decimal_formatter,
503                field,
504                &[TimeZoneFormatterUnit::ExemplarCity],
505            )?
506        }
507        (FieldSymbol::TimeZone(fields::TimeZone::LocalizedOffset), l) => perform_timezone_fallback(
508            w,
509            input,
510            datetime_names,
511            decimal_formatter,
512            field,
513            &[TimeZoneFormatterUnit::LocalizedOffset(l)],
514        )?,
515        (FieldSymbol::TimeZone(fields::TimeZone::Location), _) => perform_timezone_fallback(
516            w,
517            input,
518            datetime_names,
519            decimal_formatter,
520            field,
521            &[TimeZoneFormatterUnit::Bcp47Id],
522        )?,
523        (FieldSymbol::TimeZone(fields::TimeZone::IsoWithZ), l) => perform_timezone_fallback(
524            w,
525            input,
526            datetime_names,
527            decimal_formatter,
528            field,
529            &[TimeZoneFormatterUnit::Iso8601(Iso8601Format::with_z(l))],
530        )?,
531        (FieldSymbol::TimeZone(fields::TimeZone::Iso), l) => perform_timezone_fallback(
532            w,
533            input,
534            datetime_names,
535            decimal_formatter,
536            field,
537            &[TimeZoneFormatterUnit::Iso8601(Iso8601Format::without_z(l))],
538        )?,
539    })

542// Writes an error string for the given symbol
543fn write_value_missing(
544    w: &mut (impl writeable::PartsWrite + ?Sized),
545    field: fields::Field,
546) -> Result<(), fmt::Error> {
547    w.with_part(Part::ERROR, |w| {
548        "{".write_to(w)?;
549        char::from(field.symbol).write_to(w)?;
550        "}".write_to(w)
551    })
552}

554fn perform_timezone_fallback(
555    w: &mut (impl writeable::PartsWrite + ?Sized),
556    input: &DateTimeInputUnchecked,
557    datetime_names: &RawDateTimeNamesBorrowed,
558    fdf: Option<&DecimalFormatter>,
559    field: fields::Field,
560    units: &[TimeZoneFormatterUnit],
561) -> Result<Result<(), FormattedDateTimePatternError>, core::fmt::Error> {
562    const PART: Part = parts::TIME_ZONE_NAME;
563    let payloads = datetime_names.get_payloads();
564    let mut r = Err(FormatTimeZoneError::Fallback);
565    for unit in units {
566        let mut inner_result = None;
567        w.with_part(PART, |w| {
568            inner_result = Some(unit.format(w, input, payloads, fdf)?);
569            Ok(())
570        })?;
571        match inner_result {
572            Some(Err(FormatTimeZoneError::Fallback)) => {
573                // Expected, try the next unit
574                continue;
575            }
576            Some(r2) => {
577                r = r2;
578                break;
579            }
580            None => {
581                debug_assert!(false, "unreachable");
582                return Err(fmt::Error);
583            }
584        }
585    }
586
587    Ok(match r {
588        Ok(()) => Ok(()),
589        Err(e) => {
590            if let Some(offset) = input.zone_offset {
591                w.with_part(PART, |w| {
592                    w.with_part(Part::ERROR, |w| {
593                        Iso8601Format::without_z(field.length).format_infallible(w, offset)
594                    })
595                })?;
596            } else {
597                w.with_part(PART, |w| write_value_missing(w, field))?;
598            }
599            match e {
600                FormatTimeZoneError::DecimalFormatterNotLoaded => {
601                    Err(FormattedDateTimePatternError::DecimalFormatterNotLoaded)
602                }
603                FormatTimeZoneError::NamesNotLoaded => Err(
604                    FormattedDateTimePatternError::NamesNotLoaded(ErrorField(field)),
605                ),
606                FormatTimeZoneError::MissingInputField(kind) => {
607                    Err(FormattedDateTimePatternError::MissingInputField(kind))
608                }
609                FormatTimeZoneError::Fallback => {
610                    debug_assert!(false, "timezone fallback chain fell through {input:?}");
611                    Ok(())
612                }
613            }
614        }
615    })
616}

621mod tests {
622    use super::*;
623    use icu_decimal::options::{DecimalFormatterOptions, GroupingStrategy};
624
625    #[test]
626    fn test_format_number() {
627        let values = &[2, 20, 201, 2017, 20173];
628        let samples = &[
629            (FieldLength::One, ["2", "20", "201", "2017", "20173"]),
630            (FieldLength::Two, ["02", "20", "201", "2017", "20173"]),
631            (FieldLength::Three, ["002", "020", "201", "2017", "20173"]),
632            (FieldLength::Four, ["0002", "0020", "0201", "2017", "20173"]),
633        ];
634
635        let mut decimal_formatter_options = DecimalFormatterOptions::default();
636        decimal_formatter_options.grouping_strategy = Some(GroupingStrategy::Never);
637        let decimal_formatter = DecimalFormatter::try_new(
638            icu_locale_core::locale!("en").into(),
639            decimal_formatter_options,
640        )
641        .unwrap();
642
643        for (length, expected) in samples {
644            for (value, expected) in values.iter().zip(expected) {
645                let mut s = String::new();
646                try_write_number_without_part(
647                    &mut writeable::adapters::CoreWriteAsPartsWrite(&mut s),
648                    Some(&decimal_formatter),
649                    Decimal::from(*value),
650                    *length,
651                )
652                .unwrap()
653                .unwrap();
654                assert_eq!(s, *expected);
655            }
656        }
