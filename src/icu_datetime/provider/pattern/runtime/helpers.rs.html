10/// Helper function which takes a runtime `Pattern` and calls
11/// the callback function passing each item until the callback
12/// returns a new value.
13///
14/// When that happens, the `Pattern` is turned into an owned one,
15/// and the old item gets replaced with the new one.
16///
17/// The utility of this function is to allow for a single
18/// item to be replaced allocating the `Pattern` only if needed.
19///
20/// For a variant that replaces all matching instances, see `maybe_replace`.
21#[cfg(feature = "datagen")]
22pub fn maybe_replace_first(pattern: &mut Pattern, f: impl Fn(&PatternItem) -> Option<PatternItem>) {
23    let result = pattern
24        .items
25        .iter()
26        .enumerate()
27        .find_map(|(i, item)| f(&item).map(|result| (i, result)));
28    #[expect(clippy::indexing_slicing)] // i was produced by enumerate
29    if let Some((i, result)) = result {
30        pattern.items.to_mut_slice()[i] = result.to_unaligned();
31    }
32}

34/// Helper function which takes a runtime `Pattern` and calls
35/// the callback function passing each item.
36/// If the callback returns a new value, the old one gets
37/// replaced with it.
38///
39/// The utility of this function is to allow for a pattern
40/// to be altered, allocating the `Pattern` only if needed.
41///
42/// For a variant that replaces just the first matching instance,
43/// see `maybe_replace_first`.
44#[cfg(feature = "datagen")]
45pub fn maybe_replace(pattern: &mut Pattern, f: impl Fn(&PatternItem) -> Option<PatternItem>) {
46    let result = pattern
47        .items
48        .iter()
49        .enumerate()
50        .find_map(|(i, item)| f(&item).map(|result| (i, result)));
51    #[expect(clippy::indexing_slicing)] // i was produced by enumerate
52    if let Some((i, result)) = result {
53        let owned = pattern.items.to_mut_slice();
54        owned[i] = result.to_unaligned();
55        owned.iter_mut().skip(i).for_each(|item| {
56            if let Some(new_item) = f(&PatternItem::from_unaligned(*item)) {
57                *item = new_item.to_unaligned();
58            }
59        });
60    }
61}
