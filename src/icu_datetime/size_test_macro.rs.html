10/// Generates a test that checks the stack size of an item and a macro
11/// that should be used with `#[doc]` to document it.
12///
13/// ```text
14/// size_test!(MyType, my_type_size, 32);
15///
16/// // Add this annotation to the type's docs:
17/// #[doc = my_type_size!()]
18/// ```
19///
20/// The size should correspond to the Rust version in rust-toolchain.toml.
21///
22/// If the size on latest beta differs from rust-toolchain.toml, use the
23/// named arguments version of this macro to specify both sizes:
24///
25/// ```text
26/// size_test!(MyType, my_type_size, pinned = 32, beta = 24, nightly = 24);
27/// ```
28///
29/// The test is ignored by default but runs in CI. To run the test locally,
30/// run `cargo test -- --include-ignored`
31macro_rules! size_test {
32    ($ty:ty, $id:ident, pinned = $pinned:literal, beta = $beta:literal, nightly = $nightly:literal) => {
33        macro_rules! $id {
34            () => {
35                concat!(
36                    "\n\n",
37                    "üìè This item has a stack size of <b>",
38                    stringify!($pinned),
39                    " bytes</b> on the stable toolchain and <b>",
40                    stringify!($beta),
41                    " bytes</b> on beta toolchain at release date."
42                )
43            };
44        }
45        #[test]
46        #[cfg_attr(not(icu4x_run_size_tests), ignore)] // Doesn't work on arbitrary Rust versions
47        fn $id() {
48            let size = core::mem::size_of::<$ty>();
49            let success = match option_env!("CI_TOOLCHAIN") {
50                Some("nightly") => size == $nightly,
51                Some("beta") => size == $beta,
52                Some("pinned-stable") => size == $pinned,
53                // Manual invocation: match either size
54                _ => matches!(size, $pinned | $beta | $nightly),
55            };
56            assert!(
57                success,
58                "size_of {} = {}.\n** To reproduce this failure, run `cargo test -- --ignored` **",
59                stringify!($ty),
60                size,
61            );
62        }
63    };
64    ($ty:ty, $id:ident, $size:literal) => {
65        macro_rules! $id {
66            () => {
67                concat!(
68                    "\n\n",
69                    "üìè This item has a stack size of <b>",
70                    stringify!($size),
71                    " bytes</b> on the stable toolchain at release date."
72                )
73            };
74        }
75        #[test]
76        #[cfg_attr(not(icu4x_run_size_tests), ignore)] // Doesn't work on arbitrary Rust versions
77        fn $id() {
78            let size = core::mem::size_of::<$ty>();
79            let expected = $size;
80            assert_eq!(
81                size,
82                expected,
83                "size_of {} = {}.\n** To reproduce this failure, run `cargo test -- --ignored` **",
84                stringify!($ty),
85                size,
86            );
87        }
88    };
89}
90
91pub(crate) use size_test;
