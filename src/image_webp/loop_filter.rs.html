1//! Does loop filtering on webp lossy images
2
3#[inline]
4fn c(val: i32) -> i32 {
5    val.clamp(-128, 127)
6}
7
8//unsigned to signed
9#[inline]
10fn u2s(val: u8) -> i32 {
11    i32::from(val) - 128
12}
13
14//signed to unsigned
15#[inline]
16fn s2u(val: i32) -> u8 {
17    (c(val) + 128) as u8
18}
19
20#[inline]
21const fn diff(val1: u8, val2: u8) -> u8 {
22    if val1 > val2 {
23        val1 - val2
24    } else {
25        val2 - val1
26    }
27}
28
29//15.2
30fn common_adjust(use_outer_taps: bool, pixels: &mut [u8], point: usize, stride: usize) -> i32 {
31    let p1 = u2s(pixels[point - 2 * stride]);
32    let p0 = u2s(pixels[point - stride]);
33    let q0 = u2s(pixels[point]);
34    let q1 = u2s(pixels[point + stride]);
35
36    //value for the outer 2 pixels
37    let outer = if use_outer_taps { c(p1 - q1) } else { 0 };
38
39    let mut a = c(outer + 3 * (q0 - p0));
40
41    let b = (c(a + 3)) >> 3;
42
43    a = (c(a + 4)) >> 3;
44
45    pixels[point] = s2u(q0 - a);
46    pixels[point - stride] = s2u(p0 + b);
47
48    a
49}
50
51fn simple_threshold(filter_limit: i32, pixels: &[u8], point: usize, stride: usize) -> bool {
52    i32::from(diff(pixels[point - stride], pixels[point])) * 2
53        + i32::from(diff(pixels[point - 2 * stride], pixels[point + stride])) / 2
54        <= filter_limit
55}
56
57fn should_filter(
58    interior_limit: u8,
59    edge_limit: u8,
60    pixels: &[u8],
61    point: usize,
62    stride: usize,
63) -> bool {
64    simple_threshold(i32::from(edge_limit), pixels, point, stride)
65        // this looks like an erroneous way to compute differences between 8 points, but isn't:
66        // there are actually only 6 diff comparisons required as per the spec:
67        // https://www.rfc-editor.org/rfc/rfc6386#section-20.6
68        && diff(pixels[point - 4 * stride], pixels[point - 3 * stride]) <= interior_limit
69        && diff(pixels[point - 3 * stride], pixels[point - 2 * stride]) <= interior_limit
70        && diff(pixels[point - 2 * stride], pixels[point - stride]) <= interior_limit
71        && diff(pixels[point + 3 * stride], pixels[point + 2 * stride]) <= interior_limit
72        && diff(pixels[point + 2 * stride], pixels[point + stride]) <= interior_limit
73        && diff(pixels[point + stride], pixels[point]) <= interior_limit
74}
75
76fn high_edge_variance(threshold: u8, pixels: &[u8], point: usize, stride: usize) -> bool {
77    diff(pixels[point - 2 * stride], pixels[point - stride]) > threshold
78        || diff(pixels[point + stride], pixels[point]) > threshold
79}

81//simple filter
82//effects 4 pixels on an edge(2 each side)
83pub(crate) fn simple_segment(edge_limit: u8, pixels: &mut [u8], point: usize, stride: usize) {
84    if simple_threshold(i32::from(edge_limit), pixels, point, stride) {
85        common_adjust(true, pixels, point, stride);
86    }
87}
88
89//normal filter
90//works on the 8 pixels on the edges between subblocks inside a macroblock
91pub(crate) fn subblock_filter(
92    hev_threshold: u8,
93    interior_limit: u8,
94    edge_limit: u8,
95    pixels: &mut [u8],
96    point: usize,
97    stride: usize,
98) {
99    if should_filter(interior_limit, edge_limit, pixels, point, stride) {
100        let hv = high_edge_variance(hev_threshold, pixels, point, stride);
101
102        let a = (common_adjust(hv, pixels, point, stride) + 1) >> 1;
103
104        if !hv {
105            pixels[point + stride] = s2u(u2s(pixels[point + stride]) - a);
106            pixels[point - 2 * stride] = s2u(u2s(pixels[point - 2 * stride]) + a);
107        }
108    }
109}
110
111//normal filter
112//works on the 8 pixels on the edges between macroblocks
113pub(crate) fn macroblock_filter(
114    hev_threshold: u8,
115    interior_limit: u8,
116    edge_limit: u8,
117    pixels: &mut [u8],
118    point: usize,
119    stride: usize,
120) {
121    let mut spixels = [0i32; 8];
122    for i in 0..8 {
123        spixels[i] = u2s(pixels[point + i * stride - 4 * stride]);
124    }
125
126    if should_filter(interior_limit, edge_limit, pixels, point, stride) {
127        if !high_edge_variance(hev_threshold, pixels, point, stride) {
128            let w = c(c(spixels[2] - spixels[5]) + 3 * (spixels[4] - spixels[3]));
129
130            let mut a = c((27 * w + 63) >> 7);
131
132            pixels[point] = s2u(spixels[4] - a);
133            pixels[point - stride] = s2u(spixels[3] + a);
134
135            a = c((18 * w + 63) >> 7);
136
137            pixels[point + stride] = s2u(spixels[5] - a);
138            pixels[point - 2 * stride] = s2u(spixels[2] + a);
139
140            a = c((9 * w + 63) >> 7);
141
142            pixels[point + 2 * stride] = s2u(spixels[6] - a);
143            pixels[point - 3 * stride] = s2u(spixels[1] + a);
144        } else {
145            common_adjust(true, pixels, point, stride);
146        }
147    }
