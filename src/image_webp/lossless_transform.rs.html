24pub(crate) fn apply_predictor_transform(
25    image_data: &mut [u8],
26    width: u16,
27    height: u16,
28    size_bits: u8,
29    predictor_data: &[u8],
30) -> Result<(), DecodingError> {
31    let block_xsize = usize::from(subsample_size(width, size_bits));
32    let width = usize::from(width);
33    let height = usize::from(height);
34
35    // Handle top and left borders specially. This involves ignoring mode and using specific
36    // predictors for each.
37    image_data[3] = image_data[3].wrapping_add(255);
38    apply_predictor_transform_1(image_data, 4..width * 4, width);
39    for y in 1..height {
40        for i in 0..4 {
41            image_data[y * width * 4 + i] =
42                image_data[y * width * 4 + i].wrapping_add(image_data[(y - 1) * width * 4 + i]);
43        }
44    }
45
46    for y in 1..height {
47        for block_x in 0..block_xsize {
48            let block_index = (y >> size_bits) * block_xsize + block_x;
49            let predictor = predictor_data[block_index * 4 + 1];
50            let start_index = (y * width + (block_x << size_bits).max(1)) * 4;
51            let end_index = (y * width + ((block_x + 1) << size_bits).min(width)) * 4;
52
53            match predictor {
54                0 => apply_predictor_transform_0(image_data, start_index..end_index, width),
55                1 => apply_predictor_transform_1(image_data, start_index..end_index, width),
56                2 => apply_predictor_transform_2(image_data, start_index..end_index, width),
57                3 => apply_predictor_transform_3(image_data, start_index..end_index, width),
58                4 => apply_predictor_transform_4(image_data, start_index..end_index, width),
59                5 => apply_predictor_transform_5(image_data, start_index..end_index, width),
60                6 => apply_predictor_transform_6(image_data, start_index..end_index, width),
61                7 => apply_predictor_transform_7(image_data, start_index..end_index, width),
62                8 => apply_predictor_transform_8(image_data, start_index..end_index, width),
63                9 => apply_predictor_transform_9(image_data, start_index..end_index, width),
64                10 => apply_predictor_transform_10(image_data, start_index..end_index, width),
65                11 => apply_predictor_transform_11(image_data, start_index..end_index, width),
66                12 => apply_predictor_transform_12(image_data, start_index..end_index, width),
67                13 => apply_predictor_transform_13(image_data, start_index..end_index, width),
68                _ => {}
69            }
70        }
71    }
72
73    Ok(())
74}

141pub fn apply_predictor_transform_7(image_data: &mut [u8], range: Range<usize>, width: usize) {
142    let (old, current) = image_data[..range.end].split_at_mut(range.start);
143
144    let mut prev: [u8; 4] = old[range.start - 4..][..4].try_into().unwrap();
145    let top = &old[range.start - width * 4..][..(range.end - range.start)];
146
147    let mut current_chunks = current.chunks_exact_mut(64);
148    let mut top_chunks = top.chunks_exact(64);
149
150    for (current, top) in (&mut current_chunks).zip(&mut top_chunks) {
151        for (chunk, t) in current.chunks_exact_mut(4).zip(top.chunks_exact(4)) {
152            prev = [

162        .into_remainder()
163        .chunks_exact_mut(4)
164        .zip(top_chunks.remainder().chunks_exact(4))
165    {

214pub fn apply_predictor_transform_11(image_data: &mut [u8], range: Range<usize>, width: usize) {
215    let (old, current) = image_data[..range.end].split_at_mut(range.start);
216    let top = &old[range.start - width * 4..];
217
218    let mut l = [
219        i16::from(old[range.start - 4]),
220        i16::from(old[range.start - 3]),
221        i16::from(old[range.start - 2]),
222        i16::from(old[range.start - 1]),
223    ];
224    let mut tl = [
225        i16::from(old[range.start - width * 4 - 4]),
226        i16::from(old[range.start - width * 4 - 3]),
227        i16::from(old[range.start - width * 4 - 2]),
228        i16::from(old[range.start - width * 4 - 1]),
229    ];
230
231    for (chunk, top) in current.chunks_exact_mut(4).zip(top.chunks_exact(4)) {
232        let t = [
233            i16::from(top[0]),
234            i16::from(top[1]),
235            i16::from(top[2]),
236            i16::from(top[3]),
237        ];
238
239        let mut predict_left = 0;
240        let mut predict_top = 0;
241        for i in 0..4 {
242            let predict = l[i] + t[i] - tl[i];
243            predict_left += i16::abs(predict - l[i]);
244            predict_top += i16::abs(predict - t[i]);
245        }
246
247        if predict_left < predict_top {
248            chunk.copy_from_slice(&[
249                chunk[0].wrapping_add(l[0] as u8),
250                chunk[1].wrapping_add(l[1] as u8),
251                chunk[2].wrapping_add(l[2] as u8),
252                chunk[3].wrapping_add(l[3] as u8),
253            ]);
254        } else {
255            chunk.copy_from_slice(&[
256                chunk[0].wrapping_add(t[0] as u8),
257                chunk[1].wrapping_add(t[1] as u8),
258                chunk[2].wrapping_add(t[2] as u8),
259                chunk[3].wrapping_add(t[3] as u8),
260            ]);
261        }
262
263        tl = t;
264        l = [
265            i16::from(chunk[0]),
266            i16::from(chunk[1]),
267            i16::from(chunk[2]),
268            i16::from(chunk[3]),
269        ];
270    }
271}

343pub(crate) fn apply_color_transform(
344    image_data: &mut [u8],
345    width: u16,
346    size_bits: u8,
347    transform_data: &[u8],
348) {
349    let block_xsize = usize::from(subsample_size(width, size_bits));
350    let width = usize::from(width);
351
352    for (y, row) in image_data.chunks_exact_mut(width * 4).enumerate() {
353        let row_transform_data_start = (y >> size_bits) * block_xsize * 4;
354        // the length of block_tf_data should be `block_xsize * 4`, so we could slice it with [..block_xsize * 4]
355        // but there is no point - `.zip()` runs until either of the iterators is consumed,
356        // so the extra slicing operation would be doing more work for no reason
357        let row_tf_data = &transform_data[row_transform_data_start..];
358
359        for (block, transform) in row
360            .chunks_mut(4 << size_bits)
361            .zip(row_tf_data.chunks_exact(4))
362        {
363            let red_to_blue = transform[0];
364            let green_to_blue = transform[1];
365            let green_to_red = transform[2];
366
367            for pixel in block.chunks_exact_mut(4) {
368                let green = u32::from(pixel[1]);
369                let mut temp_red = u32::from(pixel[0]);
370                let mut temp_blue = u32::from(pixel[2]);
371
372                temp_red += color_transform_delta(green_to_red as i8, green as i8);
373                temp_blue += color_transform_delta(green_to_blue as i8, green as i8);
374                temp_blue += color_transform_delta(red_to_blue as i8, temp_red as i8);
375
376                pixel[0] = (temp_red & 0xff) as u8;
377                pixel[2] = (temp_blue & 0xff) as u8;
378            }
379        }
380    }
381}

390pub(crate) fn apply_color_indexing_transform(
391    image_data: &mut [u8],
392    width: u16,
393    height: u16,
394    table_size: u16,
395    table_data: &[u8],
396) {
397    assert!(table_size > 0);
398    if table_size > 16 {
399        // convert the table of colors into a Vec of color values that can be directly indexed
400        let mut table: Vec<[u8; 4]> = table_data
401            .chunks_exact(4)
402            // convince the compiler that each chunk is 4 bytes long, important for optimizations in the loop below
403            .map(|c| TryInto::<[u8; 4]>::try_into(c).unwrap())
404            .collect();
405        // pad the table to 256 values if it's smaller than that so we could index into it by u8 without bounds checks
406        // also required for correctness: WebP spec requires out-of-bounds indices to be treated as [0,0,0,0]
407        table.resize(256, [0; 4]);
408        // convince the compiler that the length of the table is 256 to avoid bounds checks in the loop below
409        let table: &[[u8; 4]; 256] = table.as_slice().try_into().unwrap();
410
411        for pixel in image_data.chunks_exact_mut(4) {
412            // Index is in G channel.
413            // WebP format encodes ARGB pixels, but we permute to RGBA immediately after reading from the bitstream.
414            pixel.copy_from_slice(&table[pixel[1] as usize]);
415        }
416    } else {
417        // table_size_u16 is 1 to 16
418        let table_size = table_size as u8;
419
420        // Dispatch to specialized implementation for each table size band for performance.
421        // Otherwise the compiler doesn't know the size of our copies
422        // and ends up calling out to memmove for every pixel even though a single load is sufficient.
423        if table_size <= 2 {
424            // Max 2 colors, 1 bit per pixel index -> W_BITS = 3
425            const W_BITS_VAL: u8 = 3;
426            // EXP_ENTRY_SIZE is 4 bytes/pixel * (1 << W_BITS_VAL) pixels/entry
427            const EXP_ENTRY_SIZE_VAL: usize = 4 * (1 << W_BITS_VAL); // 4 * 8 = 32
428            apply_color_indexing_transform_small_table::<W_BITS_VAL, EXP_ENTRY_SIZE_VAL>(
429                image_data, width, height, table_size, table_data,
430            );
431        } else if table_size <= 4 {
432            // Max 4 colors, 2 bits per pixel index -> W_BITS = 2
433            const W_BITS_VAL: u8 = 2;
434            const EXP_ENTRY_SIZE_VAL: usize = 4 * (1 << W_BITS_VAL); // 4 * 4 = 16
435            apply_color_indexing_transform_small_table::<W_BITS_VAL, EXP_ENTRY_SIZE_VAL>(
436                image_data, width, height, table_size, table_data,
437            );
438        } else {
439            // Max 16 colors (5 to 16), 4 bits per pixel index -> W_BITS = 1
440            // table_size_u16 must be <= 16 here
441            const W_BITS_VAL: u8 = 1;
442            const EXP_ENTRY_SIZE_VAL: usize = 4 * (1 << W_BITS_VAL); // 4 * 2 = 8
443            apply_color_indexing_transform_small_table::<W_BITS_VAL, EXP_ENTRY_SIZE_VAL>(
444                image_data, width, height, table_size, table_data,
445            );
446        }
447    }
448}

450// Helper function with const generics for W_BITS and EXP_ENTRY_SIZE
451fn apply_color_indexing_transform_small_table<const W_BITS: u8, const EXP_ENTRY_SIZE: usize>(
452    image_data: &mut [u8],
453    width: u16,
454    height: u16,
455    table_size: u8, // Max 16
456    table_data: &[u8],
457) {
458    // As of Rust 1.87 we cannot use `const` here. The compiler can still optimize them heavily
459    // because W_BITS is a const generic for each instantiation of this function.
460    let pixels_per_packed_byte_u8: u8 = 1 << W_BITS;
461    let bits_per_entry_u8: u8 = 8 / pixels_per_packed_byte_u8;
462    let mask_u8: u8 = (1 << bits_per_entry_u8) - 1;
463
464    // This is also effectively a compile-time constant for each instantiation.
465    let pixels_per_packed_byte_usize: usize = pixels_per_packed_byte_u8 as usize;
466
467    // Verify that the passed EXP_ENTRY_SIZE matches our calculation based on W_BITS, just as a sanity check.
468    debug_assert_eq!(
469        EXP_ENTRY_SIZE,
470        4 * pixels_per_packed_byte_usize,
471        "Mismatch in EXP_ENTRY_SIZE"
472    );
473
474    // Precompute the full lookup table.
475    // Each of the 256 possible packed byte values maps to an array of RGBA pixels.
476    // The array type uses the const generic EXP_ENTRY_SIZE.
477    let expanded_lookup_table_storage: Vec<[u8; EXP_ENTRY_SIZE]> = (0..256u16)
478        .map(|packed_byte_value_u16| {
479            let mut entry_pixels_array = [0u8; EXP_ENTRY_SIZE]; // Uses const generic
480            let packed_byte_value = packed_byte_value_u16 as u8;
481
482            // Loop bound is effectively constant for each instantiation.
483            for pixel_sub_index in 0..pixels_per_packed_byte_usize {
484                let shift_amount = (pixel_sub_index as u8) * bits_per_entry_u8;
485                let k = (packed_byte_value >> shift_amount) & mask_u8;
486
487                let color_source_array: [u8; 4] = if k < table_size {
488                    let color_data_offset = usize::from(k) * 4;
489                    table_data[color_data_offset..color_data_offset + 4]
490                        .try_into()
491                        .unwrap()
492                } else {
493                    [0u8; 4] // WebP spec: out-of-bounds indices are [0,0,0,0]
494                };
495
496                let array_fill_offset = pixel_sub_index * 4;
497                entry_pixels_array[array_fill_offset..array_fill_offset + 4]
498                    .copy_from_slice(&color_source_array);
499            }
500            entry_pixels_array
501        })
502        .collect();
503
504    let expanded_lookup_table_array: &[[u8; EXP_ENTRY_SIZE]; 256] =
505        expanded_lookup_table_storage.as_slice().try_into().unwrap();
506
507    let packed_image_width_in_blocks = width.div_ceil(pixels_per_packed_byte_u8.into()) as usize;
508
509    if width == 0 || height == 0 {
510        return;
511    }
512
513    let final_block_expanded_size_bytes =
514        (width as usize * 4) - EXP_ENTRY_SIZE * (packed_image_width_in_blocks.saturating_sub(1));
515
516    let input_stride_bytes_packed = packed_image_width_in_blocks * 4;
517    let output_stride_bytes_expanded = width as usize * 4;
518
519    let mut packed_indices_for_row: Vec<u8> = vec![0; packed_image_width_in_blocks];
520
521    for y_rev_idx in 0..height as usize {
522        let y = height as usize - 1 - y_rev_idx;
523
524        let packed_row_input_global_offset = y * input_stride_bytes_packed;
525        let packed_argb_row_slice =
526            &image_data[packed_row_input_global_offset..][..input_stride_bytes_packed];
527
528        for (packed_argb_chunk, packed_idx) in packed_argb_row_slice
529            .chunks_exact(4)
530            .zip(packed_indices_for_row.iter_mut())
531        {
532            *packed_idx = packed_argb_chunk[1];
533        }
534
535        let output_row_global_offset = y * output_stride_bytes_expanded;
536        let output_row_slice_mut =
537            &mut image_data[output_row_global_offset..][..output_stride_bytes_expanded];
538
539        let num_full_blocks = packed_image_width_in_blocks.saturating_sub(1);
540
541        let (full_blocks_part, final_block_part) =
542            output_row_slice_mut.split_at_mut(num_full_blocks * EXP_ENTRY_SIZE);
543
544        for (output_chunk_slice, &packed_index_byte) in full_blocks_part
545            .chunks_exact_mut(EXP_ENTRY_SIZE) // Uses const generic to avoid expensive memmove call
546            .zip(packed_indices_for_row.iter())
547        {
548            let output_chunk_array: &mut [u8; EXP_ENTRY_SIZE] =
549                output_chunk_slice.try_into().unwrap();
550
551            let colors_data_array = &expanded_lookup_table_array[packed_index_byte as usize];
552
553            *output_chunk_array = *colors_data_array;
554        }
555
556        if packed_image_width_in_blocks > 0 {
557            let final_packed_index_byte = packed_indices_for_row[packed_image_width_in_blocks - 1];
558            let colors_data_full_array =
559                &expanded_lookup_table_array[final_packed_index_byte as usize];
560
561            final_block_part
562                .copy_from_slice(&colors_data_full_array[..final_block_expanded_size_bytes]);
563        }
564    }
565}

569/// Get average of 2 bytes
570fn average2(a: u8, b: u8) -> u8 {
571    ((u16::from(a) + u16::from(b)) / 2) as u8
572}
573
574/// Clamp add subtract full on one part
575fn clamp_add_subtract_full(a: i16, b: i16, c: i16) -> u8 {
576    // Clippy suggests the clamp method, but it seems to optimize worse as of rustc 1.82.0 nightly.
577    #![allow(clippy::manual_clamp)]
578    (a + b - c).max(0).min(255) as u8
579}
580
581/// Clamp add subtract half on one part
582fn clamp_add_subtract_half(a: i16, b: i16) -> u8 {
583    // Clippy suggests the clamp method, but it seems to optimize worse as of rustc 1.82.0 nightly.
584    #![allow(clippy::manual_clamp)]
585    (a + (a - b) / 2).max(0).min(255) as u8
586}

588/// Does color transform on 2 numbers
589fn color_transform_delta(t: i8, c: i8) -> u32 {
590    (i32::from(t) * i32::from(c)) as u32 >> 5
591}
