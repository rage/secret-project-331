5//! Parsing for `TimeZoneAnnotations` and `KeyValueAnnotations`.

14        },
15        records::{Annotation, TimeZoneAnnotation},
16        timezone, Cursor,

21/// Strictly a parsing intermediary for the checking the common annotation backing.
22pub(crate) struct AnnotationSet<'a> {
23    pub(crate) tz: Option<TimeZoneAnnotation<'a>>,
24    pub(crate) calendar: Option<&'a [u8]>,
25}
26
27/// Parse a `TimeZoneAnnotation` `Annotations` set
28pub(crate) fn parse_annotation_set<'a>(
29    cursor: &mut Cursor<'a>,
30    handler: impl FnMut(Annotation<'a>) -> Option<Annotation<'a>>,
31) -> ParserResult<AnnotationSet<'a>> {
32    // Parse the first annotation.
33    let tz_annotation = timezone::parse_ambiguous_tz_annotation(cursor)?;
34
35    // Parse any `Annotations`
36    let annotations = cursor.check_or(false, is_annotation_open);
37
38    if annotations {
39        let calendar = parse_annotations(cursor, handler)?;
40        return Ok(AnnotationSet {
41            tz: tz_annotation,
42            calendar,
43        });
44    }
45
46    Ok(AnnotationSet {
47        tz: tz_annotation,
48        calendar: None,
49    })
50}

52/// Parse any number of `KeyValueAnnotation`s
53pub(crate) fn parse_annotations<'a>(
54    cursor: &mut Cursor<'a>,
55    mut handler: impl FnMut(Annotation<'a>) -> Option<Annotation<'a>>,
56) -> ParserResult<Option<&'a [u8]>> {
57    let mut calendar: Option<Annotation<'a>> = None;
58
59    while cursor.check_or(false, is_annotation_open) {
60        let annotation = handler(parse_kv_annotation(cursor)?);
61
62        match annotation {
63            // Check if the key is the registered key "u-ca".
64            Some(kv) if kv.key == "u-ca".as_bytes() => {
65                // Check the calendar
66                match calendar {
67                    Some(calendar)
68                        // if calendars do not match and one of them is critical
69                        if calendar.value != kv.value && (calendar.critical || kv.critical) =>
70                    {
71                        return Err(ParseError::CriticalDuplicateCalendar)
72                    }
73                    // If there is not yet a calendar, save it.
74                    None => {
75                        calendar = Some(kv);
76                    }
77                    _ => {}
78                }
79            }
80            Some(unknown_kv) => {
81                // Throw an error on any unrecognized annotations that are marked as critical.
82                if unknown_kv.critical {
83                    return Err(ParseError::UnrecognizedCritical);
84                }
85            }
86            None => {}
87        }
88    }
89
90    Ok(calendar.map(|a| a.value))
91}

93/// Parse an annotation with an `AnnotationKey`=`AnnotationValue` pair.
94fn parse_kv_annotation<'a>(cursor: &mut Cursor<'a>) -> ParserResult<Annotation<'a>> {
95    assert_syntax!(
96        is_annotation_open(cursor.next_or(ParseError::AnnotationOpen)?),
97        AnnotationOpen
98    );
99
100    let critical = cursor.check_or(false, is_critical_flag);
101    cursor.advance_if(critical);
102
103    // Parse AnnotationKey.
104    let annotation_key = parse_annotation_key(cursor)?;
105    assert_syntax!(
106        is_annotation_key_value_separator(cursor.next_or(ParseError::AnnotationKeyValueSeparator)?),
107        AnnotationKeyValueSeparator,
108    );
109
110    // Parse AnnotationValue.
111    let annotation_value = parse_annotation_value(cursor)?;
112    assert_syntax!(
113        is_annotation_close(cursor.next_or(ParseError::AnnotationClose)?),
114        AnnotationClose
115    );
116
117    Ok(Annotation {
118        critical,
119        key: annotation_key,
120        value: annotation_value,
121    })
122}

124/// Parse an `AnnotationKey`.
125fn parse_annotation_key<'a>(cursor: &mut Cursor<'a>) -> ParserResult<&'a [u8]> {
126    let key_start = cursor.pos();
127    assert_syntax!(
128        is_a_key_leading_char(cursor.next_or(ParseError::AnnotationKeyLeadingChar)?),
129        AnnotationKeyLeadingChar,
130    );
131
132    while let Some(potential_key_char) = cursor.next() {
133        // End of key.
134        if cursor.check_or(false, is_annotation_key_value_separator) {
135            // Return found key
136            return cursor
137                .slice(key_start, cursor.pos())
138                .ok_or(ParseError::ImplAssert);
139        }
140
141        assert_syntax!(is_a_key_char(potential_key_char), AnnotationKeyChar);
142    }
143
144    Err(ParseError::AnnotationChar)
145}
146
147/// Parse an `AnnotationValue`.
148fn parse_annotation_value<'a>(cursor: &mut Cursor<'a>) -> ParserResult<&'a [u8]> {
149    let value_start = cursor.pos();
150    cursor.advance();
151    while let Some(potential_value_char) = cursor.next() {
152        if cursor.check_or(false, is_annotation_close) {
153            // Return the determined AnnotationValue.
154            return cursor
155                .slice(value_start, cursor.pos())
156                .ok_or(ParseError::ImplAssert);
157        }
158
159        if is_hyphen(potential_value_char) {
160            assert_syntax!(
161                cursor.peek().is_some_and(is_annotation_value_component),
162                AnnotationValueCharPostHyphen,
163            );
164            cursor.advance();
165            continue;
166        }
167
168        assert_syntax!(
169            is_annotation_value_component(potential_value_char),
170            AnnotationValueChar,
171        );
172    }
173
174    Err(ParseError::AnnotationValueChar)
