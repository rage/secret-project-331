28// ==== Time Zone Annotation Parsing ====
29
30/// We support two kinds of annotations here: annotations (e.g. `[u-ca=foo]`)
31/// and "time zone annotations" (`[UTC]` or `[+05:30]`)
32///
33/// When parsing bracketed contents, we need to figure out which one we're dealing with.
34///
35/// This function returns a time zone annotation if we are dealing with a time zone,
36/// otherwise it returns None (and the caller must handle non-tz annotations).
37pub(crate) fn parse_ambiguous_tz_annotation<'a, T: EncodingType>(
38    cursor: &mut Cursor<'a, T>,
39) -> ParserResult<Option<TimeZoneAnnotation<'a, T>>> {
40    // Peek position + 1 to check for critical flag.
41    let mut current_peek = 1;
42    let critical = cursor
43        .peek_n(current_peek)?
44        .map(is_critical_flag)
45        .ok_or(ParseError::abrupt_end("AmbiguousAnnotation"))?;
46
47    // Advance cursor if critical flag present.
48    if critical {
49        current_peek += 1;
50    }
51
52    let leading_char = cursor
53        .peek_n(current_peek)?
54        .ok_or(ParseError::abrupt_end("AmbiguousAnnotation"))?;
55
56    // Ambigious start values when lowercase alpha that is shared between `TzLeadingChar` and `KeyLeadingChar`.
57    if is_a_key_leading_char(leading_char) {
58        let mut peek_pos = current_peek + 1;
59        // Go through looking for `=`
60        while let Some(ch) = cursor.peek_n(peek_pos)? {
61            if is_annotation_key_value_separator(ch) {
62                // We have an `=` sign, this is a non-tz annotation
63                return Ok(None);
64            } else if is_annotation_close(ch) {
65                // We found a `]` without an `=`, this is a time zone
66                let tz = parse_tz_annotation(cursor)?;
67                return Ok(Some(tz));
68            }
69
70            peek_pos += 1;
71        }
72        Err(ParseError::abrupt_end("AmbiguousAnnotation"))
73    } else {
74        // Unambiguously not a non-tz annotation, try parsing a tz annotation
75        let tz = parse_tz_annotation(cursor)?;
76        Ok(Some(tz))
77    }
78}

80fn parse_tz_annotation<'a, T: EncodingType>(
81    cursor: &mut Cursor<'a, T>,
82) -> ParserResult<TimeZoneAnnotation<'a, T>> {
83    assert_syntax!(
84        is_annotation_open(cursor.next_or(ParseError::AnnotationOpen)?),
85        AnnotationOpen
86    );
87
88    let critical = cursor.check_or(false, is_critical_flag)?;
89    cursor.advance_if(critical);
90
91    let tz = parse_time_zone(cursor)?;
92
93    assert_syntax!(
94        is_annotation_close(cursor.next_or(ParseError::AnnotationClose)?),
95        AnnotationClose
96    );
97
98    Ok(TimeZoneAnnotation { critical, tz })
99}

101/// Parses the [`TimeZoneIdentifier`][tz] node.
102///
103/// [tz]: https://tc39.es/proposal-temporal/#prod-TimeZoneIdentifier
104pub(crate) fn parse_time_zone<'a, T: EncodingType>(
105    cursor: &mut Cursor<'a, T>,
106) -> ParserResult<TimeZoneRecord<'a, T>> {
107    let is_iana = cursor
108        .check(is_tz_leading_char)?
109        .ok_or(ParseError::abrupt_end("TimeZoneAnnotation"))?;
110    let is_offset = cursor.check_or(false, is_ascii_sign)?;
111
112    if is_iana {
113        return Ok(TimeZoneRecord::Name(parse_tz_iana_name(cursor)?));
114    } else if is_offset {
115        let offset = parse_utc_offset_minute_precision_strict(cursor)?;
116        return Ok(TimeZoneRecord::Offset(offset));
117    }
118
119    Err(ParseError::TzLeadingChar)
120}

122/// Parse a `TimeZoneIANAName` Parse Node
123pub(crate) fn parse_tz_iana_name<'a, T: EncodingType>(
124    cursor: &mut Cursor<'a, T>,
125) -> ParserResult<&'a [T::CodeUnit]> {
126    assert_syntax!(cursor.check_or(false, is_tz_leading_char)?, TzLeadingChar);
127    let tz_name_start = cursor.pos();
128    while let Some(potential_value_char) = cursor.next()? {
129        if cursor.check_or(true, is_annotation_close)? {
130            // Return the valid TimeZoneIANAName
131            break;
132        }
133
134        if is_tz_name_separator(potential_value_char) {
135            assert_syntax!(cursor.check_or(false, is_tz_char)?, IanaCharPostSeparator,);
136            continue;
137        }
138
139        assert_syntax!(is_tz_char(potential_value_char), IanaChar,);
140    }
141
142    cursor
143        .slice(tz_name_start, cursor.pos())
144        .ok_or(ParseError::ImplAssert)
145}

149/// Parses a potentially full precision UTC offset or Z
150pub(crate) fn parse_date_time_utc_offset<T: EncodingType>(
151    cursor: &mut Cursor<T>,
152) -> ParserResult<UtcOffsetRecordOrZ> {
153    if cursor.check_or(false, is_utc_designator)? {
154        cursor.advance();
155        return Ok(UtcOffsetRecordOrZ::Z);
156    }
157
158    let utc_offset = parse_utc_offset(cursor)?;
159    Ok(UtcOffsetRecordOrZ::Offset(utc_offset))
160}
161
162/// Parse a potentially full precision `UtcOffset`
163pub(crate) fn parse_utc_offset<T: EncodingType>(
164    cursor: &mut Cursor<T>,
165) -> ParserResult<UtcOffsetRecord> {
166    let (minute_precision_offset, separated) = parse_utc_offset_minute_precision(cursor)?;
167
168    // If `UtcOffsetWithSubMinuteComponents`, continue parsing.
169    if !cursor.check_or(false, |ch| ch.is_ascii_digit() || is_time_separator(ch))? {
170        return Ok(UtcOffsetRecord::MinutePrecision(minute_precision_offset));
171    }
172
173    if Some(separated) != cursor.check(is_time_separator)? {
174        return Err(ParseError::UtcTimeSeparator);
175    }
176    cursor.advance_if(cursor.check_or(false, is_time_separator)?);
177
178    let second = parse_minute_second(cursor, false)?;
179
180    let fraction = parse_fraction(cursor)?;
181
182    Ok(UtcOffsetRecord::FullPrecisionOffset(FullPrecisionOffset {
183        minute_precision_offset,
184        second,
185        fraction,
186    }))
187}

189pub(crate) fn parse_utc_offset_minute_precision_strict<T: EncodingType>(
190    cursor: &mut Cursor<T>,
191) -> ParserResult<MinutePrecisionOffset> {
192    let (offset, _) = parse_utc_offset_minute_precision(cursor)?;
193    if cursor.check_or(false, |ch| is_time_separator(ch) || ch.is_ascii_digit())? {
194        return Err(ParseError::InvalidMinutePrecisionOffset);
195    }
196    Ok(offset)
197}

199/// Parse an `UtcOffsetMinutePrecision` node
200///
201/// Returns the offset and whether the utc parsing includes a minute separator.
202pub(crate) fn parse_utc_offset_minute_precision<T: EncodingType>(
203    cursor: &mut Cursor<T>,
204) -> ParserResult<(MinutePrecisionOffset, bool)> {
205    // https://www.rfc-editor.org/rfc/rfc3339#section-5.6
206    let sign = cursor.next_or(ParseError::abrupt_end("time-numoffset"))?;
207    if !is_ascii_sign(sign) {
208        return Err(ParseError::OffsetNeedsSign);
209    }
210    let sign = Sign::from(sign == b'+');
211
212    let hour = parse_hour(cursor)?;
213
214    // If at the end of the utc, then return.
215    if !cursor.check_or(false, |ch| ch.is_ascii_digit() || is_time_separator(ch))? {
216        let offset = MinutePrecisionOffset {
217            sign,
218            hour,
219            minute: 0,
220        };
221        return Ok((offset, false));
222    }
223    // Advance cursor beyond any TimeSeparator
224    let separated = cursor.check_or(false, is_time_separator)?;
225    cursor.advance_if(separated);
226
227    let minute = parse_minute_second(cursor, false)?;
228
229    Ok((MinutePrecisionOffset { sign, hour, minute }, separated))
230}
