5/// Supported PEM files for EC and RSA Public and Private Keys
6#[derive(Debug, PartialEq)]
7enum PemType {
8    EcPublic,
9    EcPrivate,
10    RsaPublic,
11    RsaPrivate,
12}

14#[derive(Debug, PartialEq)]
15enum Standard {
16    // Only for RSA
17    Pkcs1,
18    // RSA/EC
19    Pkcs8,
20}

24    Ec,

28/// The return type of a successful PEM encoded key with `decode_pem`
29///
30/// This struct gives a way to parse a string to a key for use in jsonwebtoken.
31/// A struct is necessary as it provides the lifetime of the key
32///
33/// PEM public private keys are encoded PKCS#1 or PKCS#8
34/// You will find that with PKCS#8 RSA keys that the PKCS#1 content
35/// is embedded inside. This is what is provided to ring via `Key::Der`
36/// For EC keys, they are always PKCS#8 on the outside but like RSA keys
37/// EC keys contain a section within that ultimately has the configuration
38/// that ring uses.
39/// Documentation about these formats is at
40/// PKCS#1: https://tools.ietf.org/html/rfc8017
41/// PKCS#8: https://tools.ietf.org/html/rfc5958
42#[derive(Debug)]
43pub(crate) struct PemEncodedKey {
44    content: Vec<u8>,
45    asn1: Vec<simple_asn1::ASN1Block>,
46    pem_type: PemType,
47    standard: Standard,
48}
49
50impl PemEncodedKey {
51    /// Read the PEM file for later key use
52    pub fn new(input: &[u8]) -> Result<PemEncodedKey> {
53        match pem::parse(input) {
54            Ok(content) => {
55                let pem_contents = content.contents;
56                let asn1_content = match simple_asn1::from_der(pem_contents.as_slice()) {
57                    Ok(asn1) => asn1,
58                    Err(_) => return Err(ErrorKind::InvalidKeyFormat.into()),
59                };
60
61                match content.tag.as_ref() {
62                    // This handles a PKCS#1 RSA Private key
63                    "RSA PRIVATE KEY" => Ok(PemEncodedKey {
64                        content: pem_contents,
65                        asn1: asn1_content,
66                        pem_type: PemType::RsaPrivate,
67                        standard: Standard::Pkcs1,
68                    }),
69                    "RSA PUBLIC KEY" => Ok(PemEncodedKey {
70                        content: pem_contents,
71                        asn1: asn1_content,
72                        pem_type: PemType::RsaPublic,
73                        standard: Standard::Pkcs1,
74                    }),
75
76                    // No "EC PRIVATE KEY"
77                    // https://security.stackexchange.com/questions/84327/converting-ecc-private-key-to-pkcs1-format
78                    // "there is no such thing as a "PKCS#1 format" for elliptic curve (EC) keys"
79
80                    // This handles PKCS#8 public & private keys
81                    tag @ "PRIVATE KEY" | tag @ "PUBLIC KEY" => match classify_pem(&asn1_content) {
82                        Some(c) => {
83                            let is_private = tag == "PRIVATE KEY";
84                            let pem_type = match c {
85                                Classification::Ec => {
86                                    if is_private {
87                                        PemType::EcPrivate
88                                    } else {
89                                        PemType::EcPublic
90                                    }
91                                }
92                                Classification::Rsa => {
93                                    if is_private {
94                                        PemType::RsaPrivate
95                                    } else {
96                                        PemType::RsaPublic
97                                    }
98                                }
99                            };
100                            Ok(PemEncodedKey {
101                                content: pem_contents,
102                                asn1: asn1_content,
103                                pem_type,
104                                standard: Standard::Pkcs8,
105                            })
106                        }
107                        None => Err(ErrorKind::InvalidKeyFormat.into()),
108                    },
109
110                    // Unknown/unsupported type
111                    _ => Err(ErrorKind::InvalidKeyFormat.into()),
112                }
113            }
114            Err(_) => Err(ErrorKind::InvalidKeyFormat.into()),
115        }
116    }
117
118    /// Can only be PKCS8
119    pub fn as_ec_private_key(&self) -> Result<&[u8]> {
120        match self.standard {
121            Standard::Pkcs1 => Err(ErrorKind::InvalidKeyFormat.into()),
122            Standard::Pkcs8 => match self.pem_type {
123                PemType::EcPrivate => Ok(self.content.as_slice()),
124                _ => Err(ErrorKind::InvalidKeyFormat.into()),
125            },
126        }
127    }
128
129    /// Can only be PKCS8
130    pub fn as_ec_public_key(&self) -> Result<&[u8]> {
131        match self.standard {
132            Standard::Pkcs1 => Err(ErrorKind::InvalidKeyFormat.into()),
133            Standard::Pkcs8 => match self.pem_type {
134                PemType::EcPublic => extract_first_bitstring(&self.asn1),
135                _ => Err(ErrorKind::InvalidKeyFormat.into()),
136            },
137        }
138    }
139
140    /// Can be PKCS1 or PKCS8
141    pub fn as_rsa_key(&self) -> Result<&[u8]> {
142        match self.standard {
143            Standard::Pkcs1 => Ok(self.content.as_slice()),
144            Standard::Pkcs8 => match self.pem_type {
145                PemType::RsaPrivate => extract_first_bitstring(&self.asn1),
146                PemType::RsaPublic => extract_first_bitstring(&self.asn1),
147                _ => Err(ErrorKind::InvalidKeyFormat.into()),
148            },
149        }
150    }
151}

153// This really just finds and returns the first bitstring or octet string
154// Which is the x coordinate for EC public keys
155// And the DER contents of an RSA key
156// Though PKCS#11 keys shouldn't have anything else.
157// It will get confusing with certificates.
158fn extract_first_bitstring(asn1: &[simple_asn1::ASN1Block]) -> Result<&[u8]> {
159    for asn1_entry in asn1.iter() {
160        match asn1_entry {
161            simple_asn1::ASN1Block::Sequence(_, entries) => {
162                if let Ok(result) = extract_first_bitstring(entries) {
163                    return Ok(result);
164                }
165            }
166            simple_asn1::ASN1Block::BitString(_, _, value) => {
167                return Ok(value.as_ref());
168            }
169            simple_asn1::ASN1Block::OctetString(_, value) => {
170                return Ok(value.as_ref());
171            }
172            _ => (),
173        }
174    }
175
176    Err(ErrorKind::InvalidEcdsaKey.into())
177}

179/// Find whether this is EC or RSA
180fn classify_pem(asn1: &[simple_asn1::ASN1Block]) -> Option<Classification> {
181    // These should be constant but the macro requires
182    // #![feature(const_vec_new)]
183    let ec_public_key_oid = simple_asn1::oid!(1, 2, 840, 10_045, 2, 1);
184    let rsa_public_key_oid = simple_asn1::oid!(1, 2, 840, 113_549, 1, 1, 1);
185
186    for asn1_entry in asn1.iter() {
187        match asn1_entry {
188            simple_asn1::ASN1Block::Sequence(_, entries) => {
189                if let Some(classification) = classify_pem(entries) {
190                    return Some(classification);
191                }
192            }
193            simple_asn1::ASN1Block::ObjectIdentifier(_, oid) => {
194                if oid == ec_public_key_oid {
195                    return Some(Classification::Ec);
196                }
197                if oid == rsa_public_key_oid {
198                    return Some(Classification::Rsa);
199                }
200            }
201            _ => {}
202        }
203    }
204    None
