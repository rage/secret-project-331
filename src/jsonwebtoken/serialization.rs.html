8pub(crate) fn b64_encode(input: &[u8]) -> String {
9    base64::encode_config(input, base64::URL_SAFE_NO_PAD)
10}
11
12pub(crate) fn b64_decode(input: &str) -> Result<Vec<u8>> {
13    base64::decode_config(input, base64::URL_SAFE_NO_PAD).map_err(|e| e.into())
14}

16/// Serializes a struct to JSON and encodes it in base64
17pub(crate) fn b64_encode_part<T: Serialize>(input: &T) -> Result<String> {
18    let json = to_string(input)?;
19    Ok(b64_encode(json.as_bytes()))
20}

22/// Decodes from base64 and deserializes from JSON to a struct AND a hashmap of Value so we can
23/// run validation on it
24pub(crate) fn from_jwt_part_claims<B: AsRef<str>, T: DeserializeOwned>(
25    encoded: B,
26) -> Result<(T, Map<String, Value>)> {
27    let s = String::from_utf8(b64_decode(encoded.as_ref())?)?;
28
29    let claims: T = from_str(&s)?;
30    let validation_map: Map<_, _> = from_str(&s)?;
31    Ok((claims, validation_map))
32}
