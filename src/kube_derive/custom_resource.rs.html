325pub(crate) fn derive(input: proc_macro2::TokenStream) -> proc_macro2::TokenStream {

327        Err(err) => return err.to_compile_error(),
328        Ok(di) => di,
329    };
330    // Limit derive to structs
331    match derive_input.data {
332        Data::Struct(_) | Data::Enum(_) => {}
333        _ => {
334            return syn::Error::new_spanned(
335                &derive_input.ident,
336                r#"Unions can not #[derive(CustomResource)]"#,
337            )
338            .to_compile_error()
339        }
340    }
341
342    let kube_attrs = match KubeAttrs::from_derive_input(&derive_input) {
343        Err(err) => return err.write_errors(),
344        Ok(attrs) => attrs,
345    };
346
347    let KubeAttrs {
348        group,
349        kind,
350        kind_struct,
351        version,
352        doc,
353        namespaced,
354        derives,
355        schema: schema_mode,
356        status,
357        plural,
358        singular,

360        shortnames,
361        printcolums,
362        selectable,
363        scale,
364        validations,
365        storage,
366        served,

370                kube_core,
371                kube,
372                k8s_openapi,
373                schemars,
374                serde,
375                serde_json,
376                std,
377            },
378        annotations,
379        labels,
380    } = kube_attrs;

382    let struct_name = kind_struct.unwrap_or_else(|| kind.clone());
383    if derive_input.ident == struct_name {
384        return syn::Error::new_spanned(
385            derive_input.ident,
386            r#"#[derive(CustomResource)] `kind = "..."` must not equal the struct name (this is generated)"#,
387        )
388        .to_compile_error();
389    }
390    let visibility = derive_input.vis;
391    let ident = derive_input.ident;

393    // 1. Create root object Foo and truncate name from FooSpec
394
395    // Default visibility is `pub(crate)`
396    // Default generics is no generics (makes little sense to re-use CRD kind?)
397    // We enforce metadata + spec's existence (always there)
398    // => No default impl
399    let rootident = Ident::new(&struct_name, Span::call_site());
400    let rootident_str = rootident.to_string();
401
402    // if status set, also add that
403    let StatusInformation {
404        field: status_field,
405        default: status_default,
406        impl_hasstatus,
407    } = process_status(&rootident, &status, &visibility, &kube_core);
408    let has_status = status.is_some();
409    let serialize_status = if has_status {
410        quote! {
411            if let Some(status) = &self.status {
412                obj.serialize_field("status", &status)?;
413            }
414        }
415    } else {
416        quote! {}
417    };
418    let has_status_value = if has_status {
419        quote! { self.status.is_some() }
420    } else {
421        quote! { false }
422    };
423
424    let mut derive_paths: Vec<Path> = vec![
425        syn::parse_quote! { #serde::Deserialize },
426        syn::parse_quote! { Clone },
427        syn::parse_quote! { Debug },
428    ];
429    let mut has_default = false;
430    for d in &derives {
431        if d == "Default" {
432            has_default = true; // overridden manually to avoid confusion
433        } else {
434            match syn::parse_str(d) {
435                Err(err) => return err.to_compile_error(),
436                Ok(d) => derive_paths.push(d),
437            }
438        }
439    }

441    // Enable schema generation by default as in v1 it is mandatory.
442    let schema_mode = schema_mode.unwrap_or(SchemaMode::Derived);
443    // We exclude fields `apiVersion`, `kind`, and `metadata` from our schema because
444    // these are validated by the API server implicitly. Also, we can't generate the
445    // schema for `metadata` (`ObjectMeta`) because it doesn't implement `JsonSchema`.
446    let schemars_skip = schema_mode.derive().then_some(quote! { #[schemars(skip)] });
447    if schema_mode.derive() && !validations.is_empty() {
448        derive_paths.push(syn::parse_quote! { #kube::KubeSchema });

458        quote! { Some(std::collections::BTreeMap::from([#((#annotations.0.to_string(), #annotations.1.to_string()),)*])) }
459    } else {
460        quote! { None }
461    };

464        quote! { Some(std::collections::BTreeMap::from([#((#labels.0.to_string(), #labels.1.to_string()),)*])) }
465    } else {
466        quote! { None }

471    let quoted_serde = Literal::string(&serde.to_token_stream().to_string());

476        #[automatically_derived]
477        #[allow(missing_docs)]
478        #[derive(#(#derive_paths),*)]
479        #[serde(rename_all = "camelCase")]
480        #[serde(crate = #quoted_serde)]
481        #schemars_attribute

673
674        let jsondata = #serde_json::json!({
675            "metadata": {
676                #crd_meta
677            },
678            "spec": {
679                "group": #group,
680                "scope": #scope,
681                "names": {
682                    "categories": categories,
683                    "plural": #plural,
684                    "singular": #name,
685                    "kind": #kind,
686                    "shortNames": shorts
687                },
688                "versions": [{
689                    "name": #version,
690                    "served": #served,
691                    "storage": #storage,
692                    #deprecation
693                    "schema": {
694                        "openAPIV3Schema": schema,
695                    },
696                    "additionalPrinterColumns": columns,
697                    #selectable
698                    "subresources": subres,
699                }],
700            }
701        });
702    };
703
704    // Implement the CustomResourceExt trait to allow users writing generic logic on top of them
705    let impl_crd = quote! {
706        impl #extver::CustomResourceExt for #rootident {
707
708            fn crd() -> #apiext::CustomResourceDefinition {
709                let columns : Vec<#apiext::CustomResourceColumnDefinition> = #serde_json::from_str(#printers).expect("valid printer column json");
710                #k8s_openapi::k8s_if_ge_1_30! {
711                    let fields : Vec<#apiext::SelectableField> = #serde_json::from_str(#fields).expect("valid selectableField column json");
712                }
713                let scale: Option<#apiext::CustomResourceSubresourceScale> = #scale;
714                let categories: Vec<String> = #serde_json::from_str(#categories_json).expect("valid categories");
715                let shorts : Vec<String> = #serde_json::from_str(#short_json).expect("valid shortnames");
716                let subres = if #has_status {
717                    if let Some(s) = &scale {
718                        #serde_json::json!({
719                            "status": {},
720                            "scale": scale
721                        })
722                    } else {
723                        #serde_json::json!({"status": {} })
724                    }
725                } else {
726                    #serde_json::json!({})
727                };
728
729                #jsondata
730                #serde_json::from_value(jsondata)
731                    .expect("valid custom resource from #[kube(attrs..)]")
732            }
733
734            fn crd_name() -> &'static str {
735                #crd_meta_name
736            }
737
738            fn api_resource() -> #kube_core::dynamic::ApiResource {
739                #kube_core::dynamic::ApiResource::erase::<Self>(&())
740            }
741
742            fn shortnames() -> &'static [&'static str] {
743                #shortnames_slice
744            }
745        }
746    };
747
748    let impl_hasspec = generate_hasspec(&ident, &rootident, &kube_core);
749
750    // Concat output
751    quote! {
752        #compile_constraints
753        #root_obj
754        #impl_resource
755        #impl_default
756        #impl_crd
757        #impl_hasspec
758        #impl_hasstatus
759    }
760}

762/// This generates the code for the `#kube_core::object::HasSpec` trait implementation.
763///
764/// All CRDs have a spec so it is implemented for all of them.
765///
766/// # Arguments
767///
768/// * `ident`: The identity (name) of the spec struct
769/// * `root ident`: The identity (name) of the main CRD struct (the one we generate in this macro)
770/// * `kube_core`: The path stream for the analagous kube::core import location from users POV
771fn generate_hasspec(spec_ident: &Ident, root_ident: &Ident, kube_core: &Path) -> TokenStream {
772    quote! {
773        impl #kube_core::object::HasSpec for #root_ident {
774            type Spec = #spec_ident;
775
776            fn spec(&self) -> &#spec_ident {
777                &self.spec
778            }
779
780            fn spec_mut(&mut self) -> &mut #spec_ident {
781                &mut self.spec
782            }
783        }
784    }
785}

794struct StatusInformation {
795    /// The code to be used for the field in the main struct
796    field: TokenStream,
797    /// The initialization code to use in a `Default` and `::new()` implementation
798    default: TokenStream,
799    /// The implementation code for the `HasStatus` trait
800    impl_hasstatus: TokenStream,
801}

803/// This processes the `status` field of a CRD.
804///
805/// As it is optional some features will be turned on or off depending on whether it's available or not.
806///
807/// # Arguments
808///
809/// * `root ident`: The identity (name) of the main CRD struct (the one we generate in this macro)
810/// * `status`: The optional name of the `status` struct to use
811/// * `visibility`: Desired visibility of the generated field
812/// * `kube_core`: The path stream for the analagous kube::core import location from users POV
813///
814/// returns: A `StatusInformation` struct
815fn process_status(
816    root_ident: &Ident,
817    status: &Option<Path>,
818    visibility: &Visibility,
819    kube_core: &Path,
820) -> StatusInformation {
821    if let Some(pth) = &status {
822        StatusInformation {
823            field: quote! {
824                #[serde(skip_serializing_if = "Option::is_none")]
825                #visibility status: Option<#pth>,
826            },
827            default: quote! { status: None, },
828            impl_hasstatus: quote! {
829                impl #kube_core::object::HasStatus for #root_ident {
830
831                    type Status = #pth;
832
833                    fn status(&self) -> Option<&#pth> {
834                        self.status.as_ref()
835                    }
836
837                    fn status_mut(&mut self) -> &mut Option<#pth> {
838                        &mut self.status
839                    }
840                }
841            },
842        }
843    } else {
844        let empty_quote = quote! {};
845        StatusInformation {
846            field: empty_quote.clone(),
847            default: empty_quote.clone(),
848            impl_hasstatus: empty_quote,
849        }
850    }
851}

853// Simple pluralizer.
854// Duplicating the code from kube (without special casing) because it's simple enough.
855// Irregular plurals must be explicitly specified.
856fn to_plural(word: &str) -> String {
857    // Words ending in s, x, z, ch, sh will be pluralized with -es (eg. foxes).
858    if word.ends_with('s')
859        || word.ends_with('x')
860        || word.ends_with('z')
861        || word.ends_with("ch")
862        || word.ends_with("sh")
863    {
864        return format!("{word}es");
865    }
866
867    // Words ending in y that are preceded by a consonant will be pluralized by
868    // replacing y with -ies (eg. puppies).
869    if word.ends_with('y') {
870        if let Some(c) = word.chars().nth(word.len() - 2) {
871            if !matches!(c, 'a' | 'e' | 'i' | 'o' | 'u') {
872                // Remove 'y' and add `ies`
873                let mut chars = word.chars();
874                chars.next_back();
875                return format!("{}ies", chars.as_str());
876            }
877        }
878    }
879
880    // All other words will have "s" added to the end (eg. days).
881    format!("{word}s")
882}
