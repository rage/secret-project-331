16/// A request to re-emit `message` at a given `Instant` (`run_at`).
17#[derive(Debug)]
18pub struct ScheduleRequest<T> {
19    pub message: T,
20    pub run_at: Instant,
21}
22
23/// Internal metadata for a scheduled message.
24struct ScheduledEntry {
25    run_at: Instant,
26    queue_key: delay_queue::Key,
27}
28
29#[pin_project(project = SchedulerProj)]
30pub struct Scheduler<T, R> {
31    /// Queue of already-scheduled messages.
32    ///
33    /// To ensure that the metadata is kept up-to-date, use `schedule_message` and
34    /// `poll_pop_queue_message` rather than manipulating this directly.
35    ///
36    /// NOTE: `scheduled` should be considered to hold the "canonical" representation of the message.
37    /// Always pull the message out of `scheduled` once it has been retrieved from `queue`.
38    queue: DelayQueue<T>,
39    /// Metadata for all currently scheduled messages. Used to detect duplicate messages.
40    ///
41    /// `scheduled` is considered to hold the "canonical" representation of the message.
42    scheduled: HashMap<T, ScheduledEntry>,
43    /// Messages that are scheduled to have happened, but have been held using `hold_unless`.
44    pending: HashSet<T>,
45    /// Incoming queue of scheduling requests.
46    #[pin]
47    requests: Fuse<R>,
48    /// Debounce time to allow for deduplication of requests. It is added to the request's
49    /// initial expiration time. If another request with the same message arrives before
50    /// the request expires, its added to the new request's expiration time. This allows
51    /// for a request to be emitted, if the scheduler is "uninterrupted" for the configured
52    /// debounce period. Its primary purpose to deduplicate requests that expire instantly.
53    debounce: Duration,
54}
55
56impl<T, R: Stream> Scheduler<T, R> {
57    fn new(requests: R, debounce: Duration) -> Self {
58        Self {
59            queue: DelayQueue::new(),
60            scheduled: HashMap::new(),
61            pending: HashSet::new(),
62            requests: requests.fuse(),
63            debounce,
64        }
65    }
66}
67
68impl<T: Hash + Eq + Clone, R> SchedulerProj<'_, T, R> {
69    /// Attempt to schedule a message into the queue.
70    ///
71    /// If the message is already in the queue then the earlier `request.run_at` takes precedence.
72    fn schedule_message(&mut self, request: ScheduleRequest<T>) {
73        if self.pending.contains(&request.message) {
74            // Message is already pending, so we can't even expedite it
75            return;
76        }
77        let next_time = request
78            .run_at
79            .checked_add(*self.debounce)
80            .map_or_else(max_schedule_time, |time|
81                // Clamp `time` to avoid [`DelayQueue`] panic (see <https://github.com/kube-rs/kube/issues/1772>)
82                time.min(max_schedule_time()));
83        match self.scheduled.raw_entry_mut().from_key(&request.message) {
84            // If new request is supposed to be earlier than the current entry's scheduled
85            // time (for eg: the new request is user triggered and the current entry is the
86            // reconciler's usual retry), then give priority to the new request.
87            RawEntryMut::Occupied(mut old_entry) if old_entry.get().run_at >= request.run_at => {
88                // Old entry will run after the new request, so replace it..
89                let entry = old_entry.get_mut();
90                self.queue.reset_at(&entry.queue_key, next_time);
91                entry.run_at = next_time;
92                old_entry.insert_key(request.message);
93            }
94            RawEntryMut::Occupied(_old_entry) => {
95                // Old entry will run before the new request, so ignore the new request..
96            }
97            RawEntryMut::Vacant(entry) => {
98                // No old entry, we're free to go!
99                let message = request.message.clone();
100                entry.insert(request.message, ScheduledEntry {
101                    run_at: next_time,
102                    queue_key: self.queue.insert_at(message, next_time),
103                });
104            }
105        }
106    }
107
108    /// Attempt to retrieve a message from the queue.
109    fn poll_pop_queue_message(
110        &mut self,
111        cx: &mut Context<'_>,
112        can_take_message: impl Fn(&T) -> bool,
113    ) -> Poll<T> {
114        if let Some(msg) = self.pending.iter().find(|msg| can_take_message(*msg)).cloned() {
115            return Poll::Ready(self.pending.take(&msg).unwrap());
116        }
117
118        loop {
119            match self.queue.poll_expired(cx) {
120                Poll::Ready(Some(msg)) => {
121                    let msg = msg.into_inner();
122                    let (msg, _) = self.scheduled.remove_entry(&msg).expect(
123                        "Expired message was popped from the Scheduler queue, but was not in the metadata map",
124                    );
125                    if can_take_message(&msg) {
126                        break Poll::Ready(msg);
127                    }
128                    self.pending.insert(msg);
129                }
130                Poll::Ready(None) | Poll::Pending => break Poll::Pending,
131            }
132        }
133    }
134
135    /// Attempt to retrieve a message from queue and mark it as pending.
136    pub fn pop_queue_message_into_pending(&mut self, cx: &mut Context<'_>) {
137        while let Poll::Ready(Some(msg)) = self.queue.poll_expired(cx) {
138            let msg = msg.into_inner();
139            self.scheduled.remove_entry(&msg).expect(
140                "Expired message was popped from the Scheduler queue, but was not in the metadata map",
141            );
142            self.pending.insert(msg);
143        }
144    }
145}

147/// See [`Scheduler::hold`]
148pub struct Hold<'a, T, R> {
149    scheduler: Pin<&'a mut Scheduler<T, R>>,
150}
151
152impl<T, R> Stream for Hold<'_, T, R>
153where
154    T: Eq + Hash + Clone,
155    R: Stream<Item = ScheduleRequest<T>>,
156{
157    type Item = T;
158
159    fn poll_next(self: Pin<&mut Self>, cx: &mut Context<'_>) -> Poll<Option<Self::Item>> {
160        let this = self.get_mut();
161        let mut scheduler = this.scheduler.as_mut().project();
162
163        loop {
164            match scheduler.requests.as_mut().poll_next(cx) {
165                Poll::Ready(Some(request)) => scheduler.schedule_message(request),
166                Poll::Ready(None) => return Poll::Ready(None),
167                Poll::Pending => break,
168            }
169        }
170
171        scheduler.pop_queue_message_into_pending(cx);
172        Poll::Pending
173    }
174}

176/// See [`Scheduler::hold_unless`]
177pub struct HoldUnless<'a, T, R, C> {
178    scheduler: Pin<&'a mut Scheduler<T, R>>,
179    can_take_message: C,
180}
181
182impl<T, R, C> Stream for HoldUnless<'_, T, R, C>
183where
184    T: Eq + Hash + Clone,
185    R: Stream<Item = ScheduleRequest<T>>,
186    C: Fn(&T) -> bool + Unpin,
187{
188    type Item = T;
189
190    fn poll_next(self: Pin<&mut Self>, cx: &mut Context<'_>) -> Poll<Option<Self::Item>> {
191        let this = self.get_mut();
192        let can_take_message = &this.can_take_message;
193        let mut scheduler = this.scheduler.as_mut().project();
194
195        loop {
196            match scheduler.requests.as_mut().poll_next(cx) {
197                Poll::Ready(Some(request)) => scheduler.schedule_message(request),
198                Poll::Ready(None) => return Poll::Ready(None),
199                Poll::Pending => break,
200            }
201        }
202
203        match scheduler.poll_pop_queue_message(cx, can_take_message) {
204            Poll::Ready(expired) => Poll::Ready(Some(expired)),
205            Poll::Pending => Poll::Pending,
206        }
207    }
208}

210impl<T, R> Scheduler<T, R>
211where
212    T: Eq + Hash + Clone,
213    R: Stream<Item = ScheduleRequest<T>>,
214{
215    /// A filtered view of the [`Scheduler`], which will keep items "pending" if
216    /// `can_take_message` returns `false`, allowing them to be handled as soon as
217    /// they are ready.
218    ///
219    /// The returned [`HoldUnless`] is designed to be short-lived: it has no allocations, and
220    /// no messages will be lost, even if it is reconstructed on each call to [`poll_next`](Self::poll_next).
221    /// In fact, this is often desirable, to avoid long-lived borrows in `can_take_message`'s closure.
222    ///
223    /// NOTE: `can_take_message` should be considered to be fairly performance-sensitive, since
224    /// it will generally be executed for each pending message, for each [`poll_next`](Self::poll_next).
225    pub fn hold_unless<C: Fn(&T) -> bool>(self: Pin<&mut Self>, can_take_message: C) -> HoldUnless<T, R, C> {
226        HoldUnless {
227            scheduler: self,
228            can_take_message,
229        }
230    }
231
232    /// A restricted view of the [`Scheduler`], which will keep all items "pending".
233    /// Its equivalent to doing `self.hold_unless(|_| false)` and is useful when the
234    /// consumer is not ready to consume the expired messages that the [`Scheduler`] emits.
235    #[must_use]
236    pub fn hold(self: Pin<&mut Self>) -> Hold<T, R> {
237        Hold { scheduler: self }
238    }
239
240    /// Checks whether `msg` is currently a pending message (held by `hold_unless`)
241    #[cfg(test)]
242    pub fn contains_pending(&self, msg: &T) -> bool {
243        self.pending.contains(msg)
244    }
245}

247impl<T, R> Stream for Scheduler<T, R>
248where
249    T: Eq + Hash + Clone,
250    R: Stream<Item = ScheduleRequest<T>>,
251{
252    type Item = T;
253
254    fn poll_next(self: Pin<&mut Self>, cx: &mut Context<'_>) -> Poll<Option<Self::Item>> {
255        Pin::new(&mut self.hold_unless(|_| true)).poll_next(cx)
256    }
257}

259/// Stream transformer that delays and deduplicates items.
260///
261/// Items are deduplicated: if an item is submitted multiple times before being emitted then it will only be
262/// emitted at the earliest `Instant`.
263///
264/// Items can be "held pending" if the item doesn't match some predicate. Items trying to schedule an item
265/// that is already pending will be discarded (since it is already going to be emitted as soon as the consumer
266/// is ready for it).
267///
268/// The [`Scheduler`] terminates as soon as `requests` does.
269pub fn scheduler<T: Eq + Hash + Clone, S: Stream<Item = ScheduleRequest<T>>>(requests: S) -> Scheduler<T, S> {
270    Scheduler::new(requests, Duration::ZERO)
271}
272
273/// Stream transformer that delays and deduplicates [`Stream`] items.
274///
275/// The debounce period lets the scheduler deduplicate requests that ask to be
276/// emitted instantly, by making sure we wait for the configured period of time
277/// to receive an uninterrupted request before actually emitting it.
278///
279/// For more info, see [`scheduler()`].
280#[allow(clippy::module_name_repetitions)]
281pub fn debounced_scheduler<T: Eq + Hash + Clone, S: Stream<Item = ScheduleRequest<T>>>(
282    requests: S,
283    debounce: Duration,
284) -> Scheduler<T, S> {
285    Scheduler::new(requests, debounce)
286}

288// [`DelayQueue`] panics when trying to schedule an event further than 2 years into the future.
289// (See <https://github.com/kube-rs/kube/issues/1772>.)
290// We limit all scheduled durations to 6 months to stay well clear of that limit.
291pub(crate) fn max_schedule_time() -> Instant {
292    Instant::now() + Duration::from_secs(86400 * 30 * 6)
293}

306        if let Poll::Ready(x) = poll {
307            x
308        } else {
309            panic!("Tried to unwrap a pending poll!")
310        }
