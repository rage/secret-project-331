1//! Partial parsers implementation of [RFC2822]: Internet Message
2//! Format.
3//!
4//! [RFC2822]: https://datatracker.ietf.org/doc/html/rfc2822
5
6use chumsky::{error::Cheap, prelude::*};
7
8use super::{rfc2234, rfc5336};
9
10// 3.2.1. Primitive Tokens
11// https://datatracker.ietf.org/doc/html/rfc2822#section-3.2.1
12
13// NO-WS-CTL       =       %d1-8 /         ; US-ASCII control characters
14//                         %d11 /          ;  that do not include the
15//                         %d12 /          ;  carriage return, line feed,
16//                         %d14-31 /       ;  and white space characters
17//                         %d127
18fn no_ws_ctl() -> impl Parser<char, char, Error = Cheap<char>> {
19    filter(|c| matches!(u32::from(*c), 1..=8 | 11 | 12 | 14..=31 | 127))
20}
21
22// text            =       %d1-9 /         ; Characters excluding CR and LF
23//                         %d11 /
24//                         %d12 /
25//                         %d14-127 /
26//                         obs-text
27fn text() -> impl Parser<char, char, Error = Cheap<char>> {
28    filter(|c| matches!(u32::from(*c), 1..=9 | 11 | 12 | 14..=127))
29}
30
31// 3.2.2. Quoted characters
32// https://datatracker.ietf.org/doc/html/rfc2822#section-3.2.2
33
34// quoted-pair     =       ("\" text) / obs-qp
35fn quoted_pair() -> impl Parser<char, char, Error = Cheap<char>> {
36    just('\\').ignore_then(text())
37}
38
39// 3.2.3. Folding white space and comments
40// https://datatracker.ietf.org/doc/html/rfc2822#section-3.2.3
41
42// FWS             =       ([*WSP CRLF] 1*WSP) /   ; Folding white space
43//                         obs-FWS
44pub(super) fn fws() -> impl Parser<char, Option<char>, Error = Cheap<char>> {
45    rfc2234::wsp()
46        .or_not()
47        .then_ignore(rfc2234::wsp().ignored().repeated())
48}
49
50// CFWS            =       *([FWS] comment) (([FWS] comment) / FWS)
51pub(super) fn cfws() -> impl Parser<char, Option<char>, Error = Cheap<char>> {
52    // TODO: comment are not currently supported, so for now a cfws is
53    // the same as a fws.
54    fws()
55}
56
57// 3.2.4. Atom
58// https://datatracker.ietf.org/doc/html/rfc2822#section-3.2.4
59
60// atext           =       ALPHA / DIGIT / ; Any character except controls,
61//                         "!" / "#" /     ;  SP, and specials.
62//                         "$" / "%" /     ;  Used for atoms
63//                         "&" / "'" /
64//                         "*" / "+" /
65//                         "-" / "/" /
66//                         "=" / "?" /
67//                         "^" / "_" /
68//                         "`" / "{" /
69//                         "|" / "}" /
70//                         "~"
71pub(super) fn atext() -> impl Parser<char, char, Error = Cheap<char>> {
72    choice((
73        rfc2234::alpha(),
74        rfc2234::digit(),
75        filter(|c| {
76            matches!(
77                *c,
78                '!' | '#'
79                    | '$'
80                    | '%'
81                    | '&'
82                    | '\''
83                    | '*'
84                    | '+'
85                    | '-'
86                    | '/'
87                    | '='
88                    | '?'
89                    | '^'
90                    | '_'
91                    | '`'
92                    | '{'
93                    | '|'
94                    | '}'
95                    | '~'
96            )
97        }),
98        // also allow non ASCII UTF8 chars
99        rfc5336::utf8_non_ascii(),
100    ))
101}
102
103// atom            =       [CFWS] 1*atext [CFWS]
104pub(super) fn atom() -> impl Parser<char, Vec<char>, Error = Cheap<char>> {
105    cfws().chain(atext().repeated().at_least(1))
106}
107
108// dot-atom        =       [CFWS] dot-atom-text [CFWS]
109pub(super) fn dot_atom() -> impl Parser<char, Vec<char>, Error = Cheap<char>> {
110    cfws().chain(dot_atom_text())
111}
112
113// dot-atom-text   =       1*atext *("." 1*atext)
114pub(super) fn dot_atom_text() -> impl Parser<char, Vec<char>, Error = Cheap<char>> {
115    atext().repeated().at_least(1).chain(
116        just('.')
117            .chain(atext().repeated().at_least(1))
118            .repeated()
119            .at_least(1)
120            .flatten(),
121    )
122}
123
124// 3.2.5. Quoted strings
125// https://datatracker.ietf.org/doc/html/rfc2822#section-3.2.5
126
127// qtext           =       NO-WS-CTL /     ; Non white space controls
128//
129//                         %d33 /          ; The rest of the US-ASCII
130//                         %d35-91 /       ;  characters not including "\"
131//                         %d93-126        ;  or the quote character
132fn qtext() -> impl Parser<char, char, Error = Cheap<char>> {
133    choice((
134        filter(|c| matches!(u32::from(*c), 33 | 35..=91 | 93..=126)),
135        no_ws_ctl(),
136    ))
137}
138
139// qcontent        =       qtext / quoted-pair
140pub(super) fn qcontent() -> impl Parser<char, char, Error = Cheap<char>> {
141    choice((qtext(), quoted_pair(), rfc5336::utf8_non_ascii()))
142}
143
144// quoted-string   =       [CFWS]
145//                         DQUOTE *([FWS] qcontent) [FWS] DQUOTE
146//                         [CFWS]
147fn quoted_string() -> impl Parser<char, Vec<char>, Error = Cheap<char>> {
148    rfc2234::dquote()
149        .ignore_then(fws().chain(qcontent()).repeated().flatten())
150        .then_ignore(text::whitespace())
151        .then_ignore(rfc2234::dquote())
152}
153
154// 3.2.6. Miscellaneous tokens
155// https://datatracker.ietf.org/doc/html/rfc2822#section-3.2.6
156
157// word            =       atom / quoted-string
158fn word() -> impl Parser<char, Vec<char>, Error = Cheap<char>> {
159    choice((quoted_string(), atom()))
160}
161
162// phrase          =       1*word / obs-phrase
163fn phrase() -> impl Parser<char, Vec<char>, Error = Cheap<char>> {
164    choice((obs_phrase(), word().repeated().at_least(1).flatten()))
165}
166
167// 3.4. Address Specification
168// https://datatracker.ietf.org/doc/html/rfc2822#section-3.4
169
170// mailbox         =       name-addr / addr-spec
171pub(crate) fn mailbox() -> impl Parser<char, (Option<String>, (String, String)), Error = Cheap<char>>
172{
173    choice((name_addr(), addr_spec().map(|addr| (None, addr))))
174        .padded()
175        .then_ignore(end())
176}
177
178// name-addr       =       [display-name] angle-addr
179fn name_addr() -> impl Parser<char, (Option<String>, (String, String)), Error = Cheap<char>> {
180    display_name().collect().or_not().then(angle_addr())
181}
182
183// angle-addr      =       [CFWS] "<" addr-spec ">" [CFWS] / obs-angle-addr
184fn angle_addr() -> impl Parser<char, (String, String), Error = Cheap<char>> {
185    addr_spec()
186        .delimited_by(just('<').ignored(), just('>').ignored())
187        .padded()
188}
189
190// display-name    =       phrase
191fn display_name() -> impl Parser<char, Vec<char>, Error = Cheap<char>> {
192    phrase()
193}
194
195// mailbox-list    =       (mailbox *("," mailbox)) / obs-mbox-list
196pub(crate) fn mailbox_list(
197) -> impl Parser<char, Vec<(Option<String>, (String, String))>, Error = Cheap<char>> {
198    choice((name_addr(), addr_spec().map(|addr| (None, addr))))
199        .separated_by(just(',').padded())
200        .then_ignore(end())
201}
202
203// 3.4.1. Addr-spec specification
204// https://datatracker.ietf.org/doc/html/rfc2822#section-3.4.1
205
206// addr-spec       =       local-part "@" domain
207pub(super) fn addr_spec() -> impl Parser<char, (String, String), Error = Cheap<char>> {
208    local_part()
209        .collect()
210        .then_ignore(just('@'))
211        .then(domain().collect())
212}
213
214// local-part      =       dot-atom / quoted-string / obs-local-part
215pub(super) fn local_part() -> impl Parser<char, Vec<char>, Error = Cheap<char>> {
216    choice((dot_atom(), quoted_string(), obs_local_part()))
217}
218
219// domain          =       dot-atom / domain-literal / obs-domain
220pub(super) fn domain() -> impl Parser<char, Vec<char>, Error = Cheap<char>> {
221    // NOTE: omitting domain-literal since it may never be used
222    choice((dot_atom(), obs_domain()))
223}

225// 4.1. Miscellaneous obsolete tokens
226// https://datatracker.ietf.org/doc/html/rfc2822#section-4.1
227
228// obs-phrase      =       word *(word / "." / CFWS)
229fn obs_phrase() -> impl Parser<char, Vec<char>, Error = Cheap<char>> {
230    // NOTE: the CFWS is already captured by the word, no need to add
231    // it there.
232    word().chain(
233        choice((word(), just('.').repeated().exactly(1)))
234            .repeated()
235            .flatten(),
236    )
237}
238
239// 4.4. Obsolete Addressing
240// https://datatracker.ietf.org/doc/html/rfc2822#section-4.4
241
242// obs-local-part  =       word *("." word)
243pub(super) fn obs_local_part() -> impl Parser<char, Vec<char>, Error = Cheap<char>> {
244    word().chain(just('.').chain(word()).repeated().flatten())
245}
246
247// obs-domain      =       atom *("." atom)
248pub(super) fn obs_domain() -> impl Parser<char, Vec<char>, Error = Cheap<char>> {
249    atom().chain(just('.').chain(atom()).repeated().flatten())
250}
