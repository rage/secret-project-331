6    debug_assert!(a.last() != Some(&0));

10    if a_len < b_len {
11        return Less;
12    }
13    if a_len > b_len {
14        return Greater;
15    }

19            return Less;
20        }
21        if ai > bi {
22            return Greater;
23        }
24    }
