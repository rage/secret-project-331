6pub fn mul_with_carry(a: BigDigit, b: BigDigit, acc: &mut DoubleBigDigit) -> BigDigit {
7    *acc += (a as DoubleBigDigit) * (b as DoubleBigDigit);
8    let lo = *acc as BigDigit;
9    *acc >>= BITS;
10    lo
