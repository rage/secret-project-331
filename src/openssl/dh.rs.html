45    pub fn check_key(&self) -> Result<bool, ErrorStack> {
46        unsafe {

54impl Dh<Params> {
55    pub fn from_params(p: BigNum, g: BigNum, q: BigNum) -> Result<Dh<Params>, ErrorStack> {
56        Self::from_pqg(p, Some(q), g)
57    }
58
59    /// Creates a DH instance based upon the given primes and generator params.
60    #[corresponds(DH_set0_pqg)]
61    pub fn from_pqg(
62        prime_p: BigNum,
63        prime_q: Option<BigNum>,
64        generator: BigNum,
65    ) -> Result<Dh<Params>, ErrorStack> {
66        unsafe {
67            let dh = Dh::from_ptr(cvt_p(ffi::DH_new())?);
68            cvt(DH_set0_pqg(
69                dh.0,
70                prime_p.as_ptr(),
71                prime_q.as_ref().map_or(ptr::null_mut(), |q| q.as_ptr()),
72                generator.as_ptr(),
73            ))?;
74            mem::forget((prime_p, prime_q, generator));
75            Ok(dh)
76        }
77    }
78
79    /// Sets the public key on the DH object.
80    pub fn set_public_key(self, pub_key: BigNum) -> Result<Dh<Public>, ErrorStack> {
81        unsafe {
82            let dh_ptr = self.0;
83            cvt(DH_set0_key(dh_ptr, pub_key.as_ptr(), ptr::null_mut()))?;
84            mem::forget((self, pub_key));
85            Ok(Dh::from_ptr(dh_ptr))
86        }
87    }
88
89    /// Sets the private key on the DH object and recomputes the public key.
90    pub fn set_private_key(self, priv_key: BigNum) -> Result<Dh<Private>, ErrorStack> {
91        unsafe {
92            let dh_ptr = self.0;
93            cvt(DH_set0_key(dh_ptr, ptr::null_mut(), priv_key.as_ptr()))?;
94            mem::forget(priv_key);
95
96            cvt(ffi::DH_generate_key(dh_ptr))?;
97            mem::forget(self);
98            Ok(Dh::from_ptr(dh_ptr))
99        }
100    }
101
102    /// Sets the public and private keys on the DH object.
103    pub fn set_key(self, pub_key: BigNum, priv_key: BigNum) -> Result<Dh<Private>, ErrorStack> {
104        unsafe {
105            let dh_ptr = self.0;
106            cvt(DH_set0_key(dh_ptr, pub_key.as_ptr(), priv_key.as_ptr()))?;
107            mem::forget((self, pub_key, priv_key));
108            Ok(Dh::from_ptr(dh_ptr))
109        }
110    }
111
112    /// Generates DH params based on the given `prime_len` and a fixed `generator` value.
113    #[corresponds(DH_generate_parameters_ex)]
114    pub fn generate_params(prime_len: u32, generator: u32) -> Result<Dh<Params>, ErrorStack> {
115        unsafe {
116            let dh = Dh::from_ptr(cvt_p(ffi::DH_new())?);
117            cvt(ffi::DH_generate_parameters_ex(
118                dh.0,
119                prime_len as i32,
120                generator as i32,
121                ptr::null_mut(),
122            ))?;
123            Ok(dh)
124        }
125    }
126
127    /// Generates a public and a private key based on the DH params.
128    #[corresponds(DH_generate_key)]
129    pub fn generate_key(self) -> Result<Dh<Private>, ErrorStack> {
130        unsafe {
131            let dh_ptr = self.0;
132            cvt(ffi::DH_generate_key(dh_ptr))?;
133            mem::forget(self);
134            Ok(Dh::from_ptr(dh_ptr))
135        }
136    }

187impl<T> Dh<T>
188where
189    T: HasParams,
190{
191    /// Returns the prime `p` from the DH instance.
192    #[corresponds(DH_get0_pqg)]
193    pub fn prime_p(&self) -> &BigNumRef {
194        let mut p = ptr::null();
195        unsafe {
196            DH_get0_pqg(self.as_ptr(), &mut p, ptr::null_mut(), ptr::null_mut());
197            BigNumRef::from_ptr(p as *mut _)
198        }
199    }
200
201    /// Returns the prime `q` from the DH instance.
202    #[corresponds(DH_get0_pqg)]
203    pub fn prime_q(&self) -> Option<&BigNumRef> {
204        let mut q = ptr::null();
205        unsafe {
206            DH_get0_pqg(self.as_ptr(), ptr::null_mut(), &mut q, ptr::null_mut());
207            if q.is_null() {
208                None
209            } else {
210                Some(BigNumRef::from_ptr(q as *mut _))
211            }
212        }
213    }
214
215    /// Returns the generator from the DH instance.
216    #[corresponds(DH_get0_pqg)]
217    pub fn generator(&self) -> &BigNumRef {
218        let mut g = ptr::null();
219        unsafe {
220            DH_get0_pqg(self.as_ptr(), ptr::null_mut(), ptr::null_mut(), &mut g);
221            BigNumRef::from_ptr(g as *mut _)
222        }
223    }
224}

241impl<T> DhRef<T>
242where
243    T: HasPrivate,
244{
245    /// Computes a shared secret from the own private key and the given `public_key`.
246    #[corresponds(DH_compute_key)]
247    pub fn compute_key(&self, public_key: &BigNumRef) -> Result<Vec<u8>, ErrorStack> {
248        unsafe {
249            let key_len = ffi::DH_size(self.as_ptr());
250            let mut key = vec![0u8; key_len as usize];
251            cvt(ffi::DH_compute_key(
252                key.as_mut_ptr(),
253                public_key.as_ptr(),
254                self.as_ptr(),
255            ))?;
256            Ok(key)
257        }
258    }
259
260    /// Returns the private key from the DH instance.
261    #[corresponds(DH_get0_key)]
262    pub fn private_key(&self) -> &BigNumRef {
263        let mut priv_key = ptr::null();
264        unsafe {
265            DH_get0_key(self.as_ptr(), ptr::null_mut(), &mut priv_key);
266            BigNumRef::from_ptr(priv_key as *mut _)
267        }
268    }
269}

294        ) {
295            if !p.is_null() {
296                *p = (*dh).p;
297            }
298            if !q.is_null() {
299                *q = (*dh).q;
300            }
301            if !g.is_null() {
302                *g = (*dh).g;
303            }
304        }

306        #[allow(bad_style)]
307        unsafe fn DH_set0_key(
308            dh: *mut ffi::DH,

310            priv_key: *mut ffi::BIGNUM,

312            (*dh).pub_key = pub_key;
313            (*dh).priv_key = priv_key;
314            1
315        }
316
317        #[allow(bad_style)]
318        unsafe fn DH_get0_key(
319            dh: *mut ffi::DH,
320            pub_key: *mut *const ffi::BIGNUM,
321            priv_key: *mut *const ffi::BIGNUM,
322        ) {
323            if !pub_key.is_null() {

325            }
326            if !priv_key.is_null() {
