23/// A numerical identifier for an OpenSSL object.
24///
25/// Objects in OpenSSL can have a short name, a long name, and
26/// a numerical identifier (NID). For convenience, objects
27/// are usually represented in source code using these numeric
28/// identifiers.
29///
30/// Users should generally not need to create new `Nid`s.
31///
32/// # Examples
33///
34/// To view the integer representation of a `Nid`:
35///
36/// ```
37/// use openssl::nid::Nid;
38///
39/// assert!(Nid::AES_256_GCM.as_raw() == 901);
40/// ```
41///
42/// # External Documentation
43///
44/// The following documentation provides context about `Nid`s and their usage
45/// in OpenSSL.
46///
47/// - [Obj_nid2obj](https://www.openssl.org/docs/manmaster/crypto/OBJ_create.html)
48#[derive(Debug, Copy, Clone, PartialEq, Eq, Hash)]
49pub struct Nid(c_int);
50
51#[allow(non_snake_case)]
52impl Nid {
53    /// Create a `Nid` from an integer representation.
54    pub const fn from_raw(raw: c_int) -> Nid {
55        Nid(raw)
56    }
57
58    /// Return the integer representation of a `Nid`.
59    #[allow(clippy::trivially_copy_pass_by_ref)]
60    pub const fn as_raw(&self) -> c_int {
61        self.0
62    }

64    /// Creates a new `Nid` for the `oid` with short name `sn` and long name `ln`.
65    #[corresponds(OBJ_create)]
66    pub fn create(oid: &str, sn: &str, ln: &str) -> Result<Nid, ErrorStack> {
67        unsafe {
68            ffi::init();
69            let oid = CString::new(oid).unwrap();
70            let sn = CString::new(sn).unwrap();
71            let ln = CString::new(ln).unwrap();
72            let raw = ffi::OBJ_create(oid.as_ptr(), sn.as_ptr(), ln.as_ptr());
73            if raw == ffi::NID_undef {
74                Err(ErrorStack::get())
75            } else {
76                Ok(Nid(raw))
77            }
78        }
79    }

81    /// Returns the `Nid`s of the digest and public key algorithms associated with a signature ID.
82    #[corresponds(OBJ_find_sigid_algs)]
83    #[allow(clippy::trivially_copy_pass_by_ref)]
84    pub fn signature_algorithms(&self) -> Option<SignatureAlgorithms> {
85        unsafe {
86            let mut digest = 0;
87            let mut pkey = 0;
88            if ffi::OBJ_find_sigid_algs(self.0, &mut digest, &mut pkey) == 1 {
89                Some(SignatureAlgorithms {
90                    digest: Nid(digest),
91                    pkey: Nid(pkey),
92                })
93            } else {
94                None
95            }
96        }
97    }

99    /// Returns the string representation of a `Nid` (long).
100    #[corresponds(OBJ_nid2ln)]
101    #[allow(clippy::trivially_copy_pass_by_ref)]
102    pub fn long_name(&self) -> Result<&'static str, ErrorStack> {
103        unsafe {
104            cvt_p(ffi::OBJ_nid2ln(self.0) as *mut c_char)
105                .map(|nameptr| str::from_utf8(CStr::from_ptr(nameptr).to_bytes()).unwrap())
106        }
107    }

116        }
117    }
