30/// A flag that allows control over the decoding strictness.
31#[derive(Debug, Clone, PartialEq, Eq, PartialOrd, Ord, Hash)]
32pub enum ParseMode {
33    /// Perform strict checking over the input, and return an error if any
34    /// input appears malformed.
35    Strict,
36    /// Perform robust parsing, and gracefully handle any malformed input. This
37    /// can result in the decoded output being different than what was intended.
38    Robust,
39}

41/// A flag that controls how to treat the input when encoding.
42#[derive(Debug, Clone, PartialEq, Eq, PartialOrd, Ord, Hash)]
43pub enum InputMode {
44    /// Treat the input as text, and don't encode CRLF pairs.
45    Text,
46    /// Treat the input as binary, and encode all CRLF pairs.
47    Binary,
48}

50/// Options to control encoding and decoding behaviour.
51#[derive(Debug, Clone, PartialEq, Eq, Hash)]
52pub struct Options {
53    /// Line length at which to wrap when encoding. Also
54    /// determines if input is valid or not when decoding in
55    /// strict mode.
56    line_length_limit: usize,
57    /// How to treat the input when encoding.
58    input_mode: InputMode,
59    /// How strict to be while decoding.
60    parse_mode: ParseMode,
61}

88/// An error type that represents different kinds of decoding errors.
89#[derive(Debug, Clone, PartialEq, Eq, PartialOrd, Ord, Hash)]
90pub enum QuotedPrintableError {
91    /// A byte was found in the input that was outside of the allowed range. The
92    /// allowed range is the horizontal tab (ASCII 0x09), CR/LF characters (ASCII
93    /// 0x0D and 0x0A), and anything in the ASCII range 0x20 to 0x7E, inclusive.
94    InvalidByte,
95    /// Lines where found in the input that exceeded 76 bytes in length, excluding
96    /// the terminating CRLF.
97    LineTooLong,
98    /// An '=' character was found in the input without the proper number of
99    /// hex-characters following it. This includes '=' characters followed
100    /// by a single character and then the CRLF pair, for example.
101    IncompleteHexOctet,
102    /// An '=' character was found with two following characters, but they were
103    /// not hex characters. '=Hi' for example would be an invalid encoding.
104    InvalidHexOctet,
105    /// An '=' character was found with two following hex characters, but the
106    /// hex characters were lowercase rather than uppercase. The spec explicitly
107    /// requires uppercase hex to be used, so this is considered an error.
108    LowercaseHexOctet,
109}
110
111impl lib::fmt::Display for QuotedPrintableError {
112    fn fmt(&self, f: &mut lib::fmt::Formatter) -> lib::fmt::Result {
113        match *self {
114            QuotedPrintableError::InvalidByte => {
115                write!(
116                    f,
117                    "A unallowed byte was found in the quoted-printable input"
118                )
119            }
120            QuotedPrintableError::LineTooLong => {
121                write!(
122                    f,
123                    "A line length in the quoted-printed input exceeded 76 bytes"
124                )
125            }
126            QuotedPrintableError::IncompleteHexOctet => {
127                write!(
128                    f,
129                    "A '=' followed by only one character was found in the input"
130                )
131            }
132            QuotedPrintableError::InvalidHexOctet => {
133                write!(
134                    f,
135                    "A '=' followed by non-hex characters was found in the input"
136                )
137            }
138            QuotedPrintableError::LowercaseHexOctet => {
139                write!(f, "A '=' was followed by lowercase hex characters")
140            }
141        }
142    }
143}

156/// Decodes a piece of quoted-printable data.
157/// This implementation is equivalent to `decode_with_options` with an
158/// `Options` that uses the provided `ParseMode`.
159#[inline(always)]
160pub fn decode<R: AsRef<[u8]>>(
161    input: R,
162    mode: ParseMode,
163) -> Result<lib::Vec<u8>, QuotedPrintableError> {
164    _decode(input.as_ref(), Options::default().parse_mode(mode))
165}

167/// Decodes a piece of quoted-printable data.
168///
169/// The quoted-printable transfer-encoding is defined in IETF RFC 2045, section
170/// 6.7. This function attempts to decode input that is conformant with that
171/// spec. Note that quoted-printable encoding is independent of charset, and so
172/// this function returns a Vec<u8> of bytes upon success. It is up to the caller
173/// to convert that to a String if desired; the charset required to do so must
174/// come from somewhere else.
175///
176/// # Examples
177///
178/// ```
179///     use quoted_printable::{decode, ParseMode};
180///     let decoded = decode("hello=3Dworld=0D=0A".as_bytes(), ParseMode::Robust).unwrap();
181///     assert_eq!("hello=world\r\n", String::from_utf8(decoded).unwrap());
182/// ```
183///
184/// # Errors
185///
186/// If this function is called with the ParseMode::Strict option, then it may return
187/// a QuotedPrintableError if it detects that the input does not strictly conform
188/// to the quoted-printable spec. If this function is called with ParseMode::Robust,
189/// then it will attempt to gracefully handle any errors that arise. This might
190/// result in input bytes being stripped out and ignored in some cases. Refer
191/// to IETF RFC 2045, section 6.7 for details on what constitutes valid and
192/// invalid input, and what a "robust" implementation would do in the face of
193/// invalid input.
194#[inline(always)]
195pub fn decode_with_options<R: AsRef<[u8]>>(
196    input: R,
197    options: Options,
198) -> Result<lib::Vec<u8>, QuotedPrintableError> {
199    _decode(input.as_ref(), options)
200}

202fn _decode(input: &[u8], options: Options) -> Result<lib::Vec<u8>, QuotedPrintableError> {
203    let filtered = input
204        .into_iter()
205        .filter_map(|&c| match c {
206            b'\t' | b'\r' | b'\n' | b' '..=b'~' => Some(c as char),
207            _ => None,
208        })
209        .collect::<lib::String>();
210    if options.parse_mode == ParseMode::Strict && filtered.len() != input.len() {
211        return Err(QuotedPrintableError::InvalidByte);
212    }
213    let mut decoded = lib::Vec::new();
214    let mut lines = filtered.lines();
215    let mut add_line_break = None;
216    loop {
217        let mut bytes = match lines.next() {
218            Some(v) => v.trim_end().bytes(),
219            None => {
220                if options.parse_mode == ParseMode::Strict && add_line_break == Some(false) {
221                    return Err(QuotedPrintableError::IncompleteHexOctet);
222                }
223                break;
224            }
225        };
226
227        if options.parse_mode == ParseMode::Strict && bytes.len() > options.line_length_limit {
228            return Err(QuotedPrintableError::LineTooLong);
229        }
230
231        if add_line_break == Some(true) {
232            decoded.push(b'\r');
233            decoded.push(b'\n');
234            add_line_break = Some(false);
235        }
236
237        loop {
238            let byte = match bytes.next() {
239                Some(v) => v,
240                None => {
241                    add_line_break = Some(true);
242                    break;
243                }
244            };
245
246            if byte == b'=' {
247                let upper = match bytes.next() {
248                    Some(v) => v,
249                    None => break,
250                };
251                let lower = match bytes.next() {
252                    Some(v) => v,
253                    None => {
254                        if options.parse_mode == ParseMode::Strict {
255                            return Err(QuotedPrintableError::IncompleteHexOctet);
256                        }
257                        decoded.push(byte);
258                        decoded.push(upper);
259                        add_line_break = Some(true);
260                        break;
261                    }
262                };
263                let upper_char = upper as char;
264                let lower_char = lower as char;
265                if upper_char.is_digit(16) && lower_char.is_digit(16) {
266                    if options.parse_mode == ParseMode::Strict {
267                        if upper_char.to_uppercase().next() != Some(upper_char)
268                            || lower_char.to_uppercase().next() != Some(lower_char)
269                        {
270                            return Err(QuotedPrintableError::LowercaseHexOctet);
271                        }
272                    }
273                    let combined =
274                        upper_char.to_digit(16).unwrap() << 4 | lower_char.to_digit(16).unwrap();
275                    decoded.push(combined as u8);
276                } else {
277                    if options.parse_mode == ParseMode::Strict {
278                        return Err(QuotedPrintableError::InvalidHexOctet);
279                    }
280                    decoded.push(byte);
281                    decoded.push(upper);
282                    decoded.push(lower);
283                }
284            } else {
285                decoded.push(byte);
286            }
287        }
288    }
289
290    if filtered.ends_with('\n') {
291        // the filtered.lines() call above ignores trailing newlines instead
292        // of returning an empty string in the last element. So if there was
293        // a trailing newline, let's tack on the CRLF to carry that through
294        // the decoder.
295        decoded.push(b'\r');
296        decoded.push(b'\n');
297    }
298
299    Ok(decoded)
300}

302fn append(
303    result: &mut lib::String,
304    to_append: &[char],
305    bytes_on_line: &mut usize,
306    limit: usize,
307    backup_pos: &mut usize,
308) {
309    if *bytes_on_line + to_append.len() > limit {
310        if *bytes_on_line == limit {
311            // We're already at the max length, so inserting the '=' in the soft
312            // line break would put us over. Instead, we insert the soft line
313            // break at the backup pos, which is just before the last thing
314            // appended.
315            *bytes_on_line = result.len() - *backup_pos;
316            result.insert_str(*backup_pos, "=\r\n");
317        } else {
318            result.push_str("=\r\n");
319            *bytes_on_line = 0;
320        }
321    }
322    result.extend(to_append);
323    *bytes_on_line = *bytes_on_line + to_append.len();
324    *backup_pos = result.len() - to_append.len();
325}

327fn encode_trailing_space_tab(
328    result: &mut lib::String,
329    bytes_on_line: &mut usize,
330    limit: usize,
331    backup_pos: &mut usize,
332) {
333    // If the last character before a CRLF was a space or tab, then encode it
334    // since "Octets with values of 9 and 32 ... MUST NOT be so represented
335    // at the end of an encoded line." We can just pop it off the end of the
336    // result and append the encoded version. The encoded version may end up
337    // getting bumped to a new line, but in that case we know that the soft
338    // line break '=' will always fit because we're removing one char before
339    // calling append.
340    match result.chars().last() {
341        Some(' ') => {
342            *bytes_on_line -= 1;
343            result.pop();
344            append(result, &['=', '2', '0'], bytes_on_line, limit, backup_pos);
345        }
346        Some('\t') => {
347            *bytes_on_line -= 1;
348            result.pop();
349            append(result, &['=', '0', '9'], bytes_on_line, limit, backup_pos);
350        }
351        _ => (),
352    };
353}

355/// Encodes some bytes into quoted-printable format, treating the input as text.
356///
357/// The quoted-printable transfer-encoding is defined in IETF RFC 2045, section
358/// 6.7. This function encodes a set of raw bytes into a format conformant with
359/// that spec. The output contains CRLF pairs as needed so that each line is
360/// wrapped to 76 characters or less (not including the CRLF).
361///
362/// # Examples
363///
364/// ```
365///     use quoted_printable::encode;
366///     let encoded = encode("hello, \u{20ac} zone!");
367///     assert_eq!("hello, =E2=82=AC zone!", String::from_utf8(encoded).unwrap());
368/// ```

370pub fn encode<R: AsRef<[u8]>>(input: R) -> lib::Vec<u8> {
371    let encoded_as_string = _encode(
372        input.as_ref(),
373        Options::default().input_mode(InputMode::Text),
374    );
375    encoded_as_string.into()
376}

401fn _encode(input: &[u8], options: Options) -> lib::String {
402    let limit = options.line_length_limit;
403    let mut result = lib::String::new();
404    let mut on_line: usize = 0;
405    let mut backup_pos: usize = 0;
406    let mut was_cr = false;
407
408    let mut it = input.iter().peekable();
409    while let Some(&byte) = it.next() {
410        if was_cr {
411            if byte == b'\n' {
412                encode_trailing_space_tab(&mut result, &mut on_line, limit, &mut backup_pos);
413                match options.input_mode {
414                    InputMode::Text => {
415                        result.push_str("\r\n");
416                        on_line = 0;
417                    }
418                    InputMode::Binary => {
419                        append(
420                            &mut result,
421                            &['=', '0', 'D'],
422                            &mut on_line,
423                            limit,
424                            &mut backup_pos,
425                        );
426                        append(
427                            &mut result,
428                            &['=', '0', 'A'],
429                            &mut on_line,
430                            limit,
431                            &mut backup_pos,
432                        );
433                    }
434                };
435                was_cr = false;
436                continue;
437            }
438            // encode the CR ('\r') we skipped over before
439            append(
440                &mut result,
441                &['=', '0', 'D'],
442                &mut on_line,
443                limit,
444                &mut backup_pos,
445            );
446        }
447        if byte == b'\r' {
448            // remember we had a CR ('\r') but do not encode it yet
449            was_cr = true;
450            continue;
451        } else {
452            was_cr = false;
453        }
454
455        // look for runs of characters that don't need encoding - this
456        // is very common (QP is normally used on ASCII text where
457        // most characters don't need encoding) and much faster than
458        // calling encode_byte on each character.  To keep this from
459        // completely reimplementing append(), only do this if we have
460        // at least 3 characters left on the line and don't try to
461        // deal with the line-ending stuff.
462        if limit - on_line >= 3 && !needs_encoding(byte) {
463            // peek ahead up to max line length and copy the run directly into the output
464            let mut run_len: usize = 1;
465            let max_run_len: usize = limit - on_line - 2;
466            debug_assert!(max_run_len >= run_len);
467
468            // add the char to result directly - safe because we know we're not at the line length limit
469            result.push(byte as char);
470
471            // look ahead for a run of characters we can put directly into the result
472            while let Some(&&next_byte) = it.peek() {
473                if run_len == max_run_len {
474                    break;
475                }
476                if needs_encoding(next_byte) {
477                    break;
478                }
479
480                run_len += 1;
481
482                // add the next char to result directly - this is safe
483                // because we're not close to the line length limit
484                result.push(next_byte as char);
485
486                // consume the byte so we don't see it again
487                it.next();
488            }
489
490            // update counters for where we are in the line and what was last appended
491            on_line += run_len;
492            backup_pos = result.len();
493
494            continue;
495        }
496
497        encode_byte(&mut result, byte, &mut on_line, limit, &mut backup_pos);
498    }
499
500    // we haven't yet encoded the last CR ('\r') so do it now
501    if was_cr {
502        append(
503            &mut result,
504            &['=', '0', 'D'],
505            &mut on_line,
506            limit,
507            &mut backup_pos,
508        );
509    } else {
510        encode_trailing_space_tab(&mut result, &mut on_line, limit, &mut backup_pos);
511    }
512
513    result
514}

516#[inline(always)]
517fn needs_encoding(c: u8) -> bool {
518    return match c {
519        b'=' => true,
520        b'\t' | b' '..=b'~' => false,
521        _ => true,
522    };
523}

592#[inline]
593fn encode_byte(
594    result: &mut lib::String,
595    to_append: u8,
596    on_line: &mut usize,
597    limit: usize,
598    backup_pos: &mut usize,
599) {
600    match to_append {
601        b'=' => append(result, &['=', '3', 'D'], on_line, limit, backup_pos),
602        b'\t' | b' '..=b'~' => append(result, &[char::from(to_append)], on_line, limit, backup_pos),
603        _ => append(
604            result,
605            &hex_encode_byte(to_append),
606            on_line,
607            limit,
608            backup_pos,
609        ),
610    }
611}

613#[inline(always)]
614fn hex_encode_byte(byte: u8) -> [char; 3] {
615    [
616        '=',
617        lower_nibble_to_hex(byte >> 4),
618        lower_nibble_to_hex(byte),
619    ]
620}

622#[inline(always)]
623fn lower_nibble_to_hex(half_byte: u8) -> char {
624    HEX_CHARS[(half_byte & 0x0F) as usize]
625}

769        assert_eq!(

786        );
787        // Test that soft line breaks don't break up an encoded octet
788        assert_eq!(
